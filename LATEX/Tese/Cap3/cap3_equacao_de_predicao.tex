\section{Prediction Equation}

Let's express the relationship between the vectors

\begin{equation}
	\mathbf{\hat{u}}[k] = \left[
		\begin{matrix}
			\hat{u}[k | k] \\ 
			\hat{u}[k + 1 | k] \\ 
			\vdots \\ 
			\hat{u}[k + N-1 | k]
		\end{matrix}
	\right],
	\mathbf{\hat{x}}[k] = 
		\left[
		\begin{matrix}
			\hat{x}[k + 1 | k] \\ 
			\hat{x}[k + 2 | k] \\ 
			\vdots \\ 
			\hat{x}[k + N | k]
		\end{matrix}
	\right]
\end{equation}

\noindent
based on the prediction equation

\noindent
\begin{equation}
	\label{cap3_eq_pred1}
	\hat{x}[k + i +1|k] = A\hat{x}[k+i|k] + B \hat{u}[k+i|k], 
	i = 0, 1, \ldots, N 
\end{equation}

\noindent
and initial condition \highlight{$\hat{x}[k|k] = x[k]$}.

From \eqref{cap3_eq_pred1}, it follows that

\begin{equation}
	\hat{x}[k+i|k] = A^i\hat{x}[k|k] + \sum_{j=0}^{i-1} A^{i - 1 -j} B \hat{u}[k+j|k]
\end{equation}

\noindent
which can be written for \highlight{$i = 1,2,...,N$} as

\noindent
\begin{align}
	\hat{x}[k+1|k] &= A x[k] +  B \hat{u}[k|k] \\
	\hat{x}[k+2|k] &= A^2 x[k] + A B \hat{u}[k|k] +  B \hat{u}[k+1|k] \\
	&\vdots \\
	\hat{x}[k+N|k] &= A^N x[k] + A^{N - 1} B \hat{u}[k|k] + \cdots + A B \hat{u}[k +N - 2|k] +  B \hat{u}[k + N - 1|k]
\end{align}

\noindent
or

\noindent
\begin{equation}
	\left[
		\begin{matrix}
			\hat{x}[k+1|k] \\
			\hat{x}[k+2|k] \\
			\vdots \\
			\hat{x}[k+N|k] \\
		\end{matrix}
	\right] = 
	\left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	x(0) + 
	\left[
		\begin{matrix}
			B & 0 & \cdots & 0 \\
			AB & B & \cdots & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \cdots & B
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			\hat{u}[k|k] \\
			\hat{u}[k+1|k]\\
			\vdots \\
			\hat{u}[k+N-1|k]
		\end{matrix}
	\right]
\end{equation}

This can be written in a shorter form as



\noindent
\begin{equation}
	\label{cap3_eq_predicao_xk}
	\mathbf{\hat{x}}[k] = \mathbf{A}x[k] + \mathbf{B \hat{u}}[k]
\end{equation}

\noindent
with 

\noindent
\begin{equation}
	\mathbf{A} = \left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	, 
	\mathbf{B} = \left[
		\begin{matrix}
			B  & 0 & \cdots & 0 \\
			AB & B & \cdots & 0 \\
			\vdots  & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \cdots & B
		\end{matrix}
	\right]
\end{equation}

The vectors \highlight{$\mathbf{\hat{x}}[k]$} and \highlight{$\mathbf{\hat{u}}[k]$} are linked by 
\eqref{cap3_eq_predicao_xk}, so the constraint \eqref{cap3_sx_x_le_bx} can be rewritten as

\noindent
\begin{equation}
	\mathbf{S_x} \mathbf{B} \mathbf{u} \le \mathbf{b_x} - \mathbf{S_x} \mathbf{A} x(0)
\end{equation}

Combined with \highlight{$\mathbf{S_u} \mathbf{u}[k] \le \mathbf{b_u}$}, it can be expressed in the form

\noindent
\begin{equation}
	\mathbf{S} \mathbf{u}[k] \le \mathbf{b}
\end{equation}

\noindent
with

\noindent
\begin{equation}
	\mathbf{S} = \left[ \begin{matrix} \mathbf{S_x} \mathbf{B} \\ \mathbf{S_u} \end{matrix} \right], \mathbf{b} = 
	\left[ \begin{matrix} \mathbf{b_x} - \mathbf{S_x} \mathbf{A} \hat{x}[k|k] \\ \mathbf{b_u} \end{matrix} \right]
\end{equation}
