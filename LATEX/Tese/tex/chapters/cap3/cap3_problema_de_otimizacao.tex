
\section{Optimization Problem}

The previous sections provide the two elements needed for the finite-horizon MPC problem: the condensed prediction model and the lifted constraints. We can now rewrite the cost function as

\noindent
\begin{equation}
	J = \norma{x[k|k]}{P} + \green{\overbrace{\norma{\mathbf{\hat{x}}[k]}{\mathbf{Q}}}^{\square}} + \norma{\mathbf{\hat{u}}[k]}{\mathbf{R}}
\end{equation}

\noindent
\begin{align}
	\green{\square} 
	&= \green{\norma{\mathbf{\hat{x}}[k]}{\mathbf{Q}}} \\
	&= \green{\Big[ \mathbf{A}x[k|k] + \mathbf{B \hat{u}}[k] \Big]^T \mathbf{Q} \Big[ \mathbf{A}x[k|k] + \mathbf{B \hat{u}}[k] \Big]} \\
	&= \green{\Big[ \mathbf{A}x[k|k] + \mathbf{B \hat{u}}[k] \Big]^T \mathbf{Q} \Big[ \mathbf{A}x[k|k] + \mathbf{B \hat{u}}[k] \Big]} \\
	\begin{split}
		&= \green{\norma{\mathbf{A}x[k|k]}{\mathbf{Q}} + \Big[ \mathbf{A}x[k|k] \Big]^T \mathbf{Q} \mathbf{B \hat{u}}[k]} \\
		&\qquad \green{ + \mathbf{\hat{u}}[k]^T \mathbf{B}^T \mathbf{Q} \mathbf{A}x[k|k] + \mathbf{\hat{u}}[k]^T \mathbf{B}^T \mathbf{Q} \mathbf{B \hat{u}}[k]}
	\end{split}
\end{align}

\noindent
replacing $\green{\square}$ back in the cost equation and reordering the elements 

\noindent
\begin{align}
	J(\mathbf{\hat{x}}, \mathbf{\hat{u}}[k]) 
	&= \purple{ \norma{x[k|k]}{P} + \norma{\mathbf{A}x[k|k]}{\mathbf{Q}} } \\
	&\quad + \orange{\Big[\mathbf{A}x[k|k]\Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}[k]	+ \mathbf{\hat{u}}[k]^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x[k|k]} \\
	&\quad + \blue{\mathbf{\hat{u}}[k]^T \mathbf{R} \mathbf{\hat{u}}[k] + \mathbf{\hat{u}}[k]^T \mathbf{B}^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}[k]}
\end{align}

The \purple{constant part} will be denoted by \purple{$c_0$}

Recalling that the matrix \highlight{$\mathbf{Q}$} is symmetric, we have:

\begin{equation}
	\orange{\Big[ \mathbf{A}x[k|k] \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}[k]} = 
	\orange{\Big\{ \mathbf{\hat{u}}[k]^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x[k|k] \Big\}^T}
\end{equation}

As these terms are scalars, it can be concluded that they are equal to each other, therefore:

\noindent
\begin{equation}
	\begin{gathered}
		\orange{\Big[ \mathbf{A}x[k|k] \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}[k] + \mathbf{\hat{u}}[k]^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x(0)} \\
		= \orange{2\Big[ \mathbf{A}x[k|k] \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}[k]}
	\end{gathered}
\end{equation}

Finally, it can be written

\noindent
\begin{equation}
	\begin{gathered}
		J(\mathbf{\hat{u}}[k]) = 
			\blue{\mathbf{\hat{u}}[k]^T \Big(\mathbf{R} + \mathbf{B}^T \mathbf{Q} \mathbf{B} \Big) \mathbf{\hat{u}}[k]} + 
			\orange{2 \Big[ \mathbf{A}x[k|k] \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}[k]} + 
			\purple{c_0}
	\end{gathered}
\end{equation}

with

\noindent
\begin{equation}
	\purple{ c_0 } = 
	\purple{ \norma{x(0)}{P} + \norma{\mathbf{A} x(0)}{\mathbf{Q}} }
\end{equation}

\purple{
	\noindent
	\begin{equation}
		c_0 = 
		\norma{x[k|k]}{P} + \norma{\mathbf{A} x[k|k]}{\mathbf{Q}}
		\nonumber
	\nonumber
	\end{equation}
}

Note that we will now use the notation \highlight{$J(\mathbf{\hat{u}}[k])$} instead of \highlight{$J(\mathbf{\hat{x}}[k], \mathbf{\hat{u}}[k])$}, 
since \highlight{$\mathbf{\hat{x}}[k]$} has been eliminated from the cost expression. 


To finalize the restructuring of the optimization problem in terms of \highlight{$\mathbf{\hat{u}}[k]$}, the relation between 
\highlight{$\mathbf{\hat{x}}[k]$} and \highlight{$\mathbf{\hat{u}}[k]$} in the restrictions must be considered.

The optimization problem can be expressed as

\begin{equation}
	\min_{\mathbf{\hat{x}}[k] \in \mathbb{R}^{nN}, \mathbf{\hat{u}}[k] \in \mathbb{R}^{pN}} J(\mathbf{\hat{x}}[k],\mathbf{\hat{u}}[k])
\end{equation}

\noindent
where

\noindent
\begin{equation}
    J(\mathbf{\hat{x}}[k], \mathbf{\hat{u}}[k]) = \norma{x[k|k]}{P} + \norma{\mathbf{\hat{x}}[k]}{\mathbf{Q}} + \norma{\mathbf{\hat{u}}[k]}{\mathbf{R}}
\end{equation}

\noindent
with

\noindent
\begin{equation}
    \mathbf{\hat{u}}[k] = \left[ \begin{matrix} \hat{u}[k|k] \\ \hat{u}[k + 1|k] \\ \vdots \\ \hat{u}[k + N - 1|k] \end{matrix} \right] \in \mathbb{R}^{pN},
    \mathbf{\hat{x}}[k] = \left[ \begin{matrix} \hat{x}[k + 1|k] \\ \hat{x}[k + 2|k] \\ \vdots \\ \hat{x}[k + N|k] \end{matrix} \right] \in \mathbb{R}^{nN},
\end{equation}

\noindent
subject to

\noindent
\begin{equation}
    \mathbf{S_x} \mathbf{\hat{x}}[k] \leq \mathbf{b_x}
\end{equation}

\noindent
\begin{equation}
    \mathbf{S_u} \mathbf{\hat{u}}[k] \leq \mathbf{b_u}
\end{equation}

Thus, the cost can be rewritten as

    \noindent
    \begin{gather}
        J(\mathbf{\hat{u}}[k]) = \mathbf{\hat{u}}[k]^T(\mathbf{R} + \mathbf{B}^T \mathbf{Q} \mathbf{B} )\mathbf{\hat{u}}[k] \nonumber \\ 
        2 \Big[ \mathbf{A} \hat{x}[k|k] \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}[k] + c_0 \nonumber
    \end{gather}

\noindent
with

\noindent
\begin{equation}
    c_0 = \norma{\hat{x}[k|k] }{P} + \norma{\mathbf{A} \hat{x}[k|k] }{\mathbf{Q}} 
    \nonumber
\end{equation}

\noindent
and the constraints can be expressed as

\noindent
\begin{equation}
    \label{eq:cap3:optimization-constraints}
    \mathbf{S} \mathbf{\hat{u}}[k] \leq \mathbf{b}
\end{equation}

Having obtained the solution \highlight{$\mathbf{\hat{u}^*}[k|k]$} for the problem 
\highlight{$\mathbb{P}\Big(\hat{x}[k|k]\Big)$}, the control to be applied to the plant is given by

\noindent
\begin{equation}
    \hat{u}[k] = \hat{u}^*[k|k]
    \nonumber
\end{equation}

\noindent
where \highlight{$\hat{u}^*[k|k]$} corresponds to the first \highlight{$p$} elements of \highlight{$\mathbf{\hat{u}}^*[k|k]$}.
