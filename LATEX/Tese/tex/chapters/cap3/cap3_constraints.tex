
\section{Constraints}

In view of \eqref{eq:cap3:state-constraint-set} and \eqref{eq:cap3:input-constraint-set}, the constraints on the optimization 
variables can be expressed as

\noindent
\begin{align}
	\mathbf{\hat{u}}[k]&= \left[
		\begin{matrix}
			\hat{u}[k | k] \\
			\hat{u}[k + 1 | k] \\
			\vdots \\
			\hat{u}[k + N-1 | k]
		\end{matrix}
	\right]
	\in \mathcal{U} \times  \mathcal{U} \times \cdots \times \mathcal{U} = \mathcal{U}^N \\
		\mathbf{\hat{x}}[k] &= \left[
		\begin{matrix}
			\hat{x}[k + 1 | k] \\
			\hat{x}[k + 2 | k] \\
			\vdots \\
			\hat{x}[k + N | k]
		\end{matrix}
	\right]
	\in \mathcal{X} \times  \mathcal{X} \times \cdots \times \mathcal{X}_f = \mathcal{X}^{N-1} \times \mathcal{X}_f
\end{align}

\noindent
where

\noindent
\begin{equation}
	\mathcal{X}_f = \big\{x \in \mathbb{R}^n : S_f x \le b_f \big\}
\end{equation}
is the maximum invariant and admissible set (MAS) concerning the terminal state and control
constraints under the infinite-horizon DLQR control law.

The constraint on \highlight{$\mathbf{\hat{u}}[k]$} can be  written as

\noindent
\begin{equation}
	\left[
		\begin{matrix}
			S_u & 0 & \cdots & 0 \\
			0 & S_u & \cdots & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \cdots & S_u
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			\hat{u}[k | k] \\
			\hat{u}[k + 1 | k] \\
			\vdots \\
			\hat{u}[k + N-1 | k]
		\end{matrix}
	\right]
	\leq
	\left[
		\begin{matrix}
			b_u \\
			b_u \\
			\vdots \\
			b_u
		\end{matrix}
	\right]
\end{equation}

\noindent
or, in a shorter form

\noindent
\begin{equation}
	\mathbf{S_u \hat{u}}[k] \leq \mathbf{b_u}
\end{equation}

\noindent
with

\noindent
\begin{equation}
	\mathbf{S_u} = diag_N(S_u), \mathbf{b_u} = col_N(b_u)
\end{equation}

Similarly, in light of \eqref{eq:cap3:polyhedral-state-constraint}, the constraints on \highlight{$\mathbf{\hat{x}}[k]$} can be written as

\begin{equation}
	\left[
		\begin{matrix}
			S_x & 0 & \cdots & 0 \\
			0 & S_x & \cdots & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \cdots & S_x
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			\hat{x}[k + 1 | k] \\
			\hat{x}[k + 2 | k] \\
			\vdots \\
			\hat{x}[k + N | k]
		\end{matrix}
	\right]
	\leq
	\left[
		\begin{matrix}
			b_x \\
			b_x \\
			\vdots \\
			b_x
		\end{matrix}
	\right]
\end{equation}

\noindent
or, in short

\noindent
\begin{equation}
	\label{eq:cap3:prediction-state-constraint}
	\mathbf{S_x}\mathbf{\hat{x}}[k] \leq \mathbf{b_x}
\end{equation}

\noindent
with

\noindent
\begin{equation}
	\mathbf{S_x} = 
	\left[
		\begin{matrix}
			diag_{N-1}(S_x) & 0 \\
			0 & S_x
		\end{matrix}
	\right], \mathbf{b_x} = 
	\left[
		\begin{matrix}
			col_{N-1}(b_x) \\
			b_f
		\end{matrix}
	\right]
\end{equation}
