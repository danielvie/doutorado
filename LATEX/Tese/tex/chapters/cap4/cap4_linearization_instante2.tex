Repeating the steps for $x(t_2)$ we have


\noindent
\begin{align}
	\label{eq:cap4:linearization-x2}
	x(t_2)  &= e^{A_2(t_2-t_1)}x(t_1) + \intlinAA{1}{2}{}{} \\
	\label{eq:cap4:linearization-reference-x2}
	\bar{x}(\bar{t}_2) &= e^{A_2(\bar{t}_2-\bar{t}_1)}\bar{x}(\bar{t}_1) + \intlinAbtt{1}{2}
\end{align}

\noindent 
subtracting \eqref{eq:cap4:linearization-reference-x2} from \eqref{eq:cap4:linearization-x2} 

\begin{equation}
	\label{eq:cap4:linearization-error-x1-x2}
	\begin{split}
		x(t_2) - \bar{x}(\bar{t}_2) = 
		\green{\overbrace{e^{A_2(t_2-t_1)}x(t_1) - e^{A_2(\bar{t}_2-\bar{t}_1)}\bar{x}(\bar{t}_1)}^{\square}} \\ 
		+ \orange{\underbrace{\intlinAA{1}{2}{}{} - \intlinAbtt{1}{2}}_{\bigcirc}}
	\end{split}
\end{equation}

\noindent 
expanding $\green{\square}$ in \eqref{eq:cap4:linearization-error-x1-x2}:

% 2.10
\noindent
\begin{align}
	\green{ \square } &=
	\green{
		e^{A_2(\bar{t}_2 + \delta t_2 - \bar{t}_1 - \delta t_1)} x(t_1)
		- e^{A_2(\bar{t}_2 - \bar{t}_1)} \bar{x}(\bar{t}_1)
	} \nonumber \\
	&= \green{
		\blue{\Fri{1}{2}} \purple{e^{A_2(\delta t_2 - \delta t_1)}} x(t_1)
		- \blue{\Fri{1}{2}} \bar{x}(\bar{t}_1)
	} \nonumber \\
	&= \green{
		\blue{\bar{F}_2} \purple{[I + A_2(\delta t_2 - \delta t_1)]} x(t_1)
		- \blue{\bar{F}_2} \bar{x}(\bar{t}_1)
	} \nonumber \\
	\label{eq:cap4:linearization-x2-green-term}
	&= \green{
		\bar{F}_2 [x(t_1) - \bar{x}(\bar{t}_1)] + \bar{F}_2 A_2 x(t_1) \delta t_2
		- \bar{F}_2 A_2 x(t_1) \delta t_1
	}
\end{align}

\noindent 
and expanding $\orange{\bigcirc}$:

% 2.11
\noindent
\begin{align}
	\orange{\bigcirc} 	&= \orange{\intlinAA{1}{2}{}{} - \intlinAbtt{1}{2}} \nonumber \\
					   	&= \orange{ \intlinCb{\bar{t}_1+\delta t_1}{\bar{t}_2+\delta t_2}{1}{2} - \intlinAbtt{1}{2}} \nonumber \\
						\begin{split}
							&= \orange{ \purple{ e^{A_2 \delta t_2} }
								\bigg[
									\blue{\intlinAbtt{1}{2}} - \intlinBb{\bar{t}_1}{\bar{t}_1 + \delta t_1}{1}{2} + \intlinBb{\bar{t}_2}{\bar{t}_2 + \delta t_2}{1}{2} \bigg] } \\
							& \qquad \orange{ - \blue{\intlinAbtt{1}{2}} }
						\end{split} \nonumber \\
						&= \orange{ \purple{(I + A_2 \delta t_2)}
							\bigg[
								\blue{\bar{G}_2} - \pink{\intlinBb{\bar{t}_1}{\bar{t}_1 + \delta t_1}{1}{2}} + \brown{\intlinBb{\bar{t}_2}{\bar{t}_2 + \delta t_2}{1}{2}}
							\bigg] - \blue{\bar{G}_2}
						} \nonumber \\
						&= \orange{ (I + A_2 \delta t_2)
							\bigg[
								\bar{G}_2 - \pink{e^{A_2(\bar{t}_2 - \bar{t}_1)} B_2 \delta t_1} + \brown{B_2 \delta t_2}
							\bigg] - \bar{G}_2
						} \nonumber \\
						&= \orange{ (I + A_2 \delta t_2)
							\bigg[
								\bar{G}_2 - \pink{\bar{F}_2 B_2 \delta t_1} + B_2 \delta t_2
							\bigg] - \bar{G}_2
						} \nonumber \\
						&= \orange{ \cancel{\bar{G}_2} - \bar{F}_2 B_2 \delta t_1 + B_2 \delta t_2 + A_2 \bar{G}_2 \delta t_2 - \cancel{\bar{G}_2} } \nonumber \\
						\label{eq:cap4:linearization-x2-orange-term}
						&= \orange{ A_2 \bar{G}_2 \delta t_2 + B_2 \delta t_2 - \bar{F}_2 B_2 \delta t_1 }
\end{align}

\noindent 
and replacing \eqref{eq:cap4:linearization-x2-green-term} and \eqref{eq:cap4:linearization-x2-orange-term} back into equation \eqref{eq:cap4:linearization-error-x1-x2} we get to the difference equation for 
$x(t_2) - \bar{x}(\bar{t}_2)$

\noindent
\begin{equation}
	\label{eq:cap4:linearization-x2-x1-recurrence}
	\begin{split}
		x(t_2) - \bar{x}(\bar{t}_2) =  & \bar{F}_2 [\green{x(t_1) - \bar{x}_1(\bar{t}_1)}] \\
							     & \quad - \bar{F}_2 [A_2 \blue{x(t_1)} + B_2] \delta t_1 \\
							     & \quad + [ \bar{F}_2 A_2 \blue{\underbrace{x(t_1)}_{\triangle}} + A_2 \bar{G}_2 + B_2] \delta t_2
	\end{split} 
\end{equation}

\noindent 
from \eqref{eq:cap4:linearization-x2-x1-recurrence}, $\blue{\triangle}$ can be expanded by replacing the resulting equation of \eqref{eq:cap4:linearization-error-x1-x0-result}

\noindent
\begin{equation}
	\label{eq:cap4:linearization-x2-reference-triangle-term}
	\blue{\triangle} = x(t_1) = \bar{x}(\bar{t}_1) + \bar{F}_1 [ x(t_0) - \bar{x}(t_0) ] + [ \bar{F}_1 A_1 x(t_0) + A_1 \bar{G}_1 + B_1 ] \delta t_1 
\end{equation}

\noindent 
note that when $\blue{\triangle}$ is replaced in eq. \eqref{eq:cap4:linearization-x2-x1-recurrence} it will produce a product with $\delta t_1$ and $\delta t_2$ like

\noindent
\begin{equation}
	x(t_1)\delta t_1 = \bar{x}(\bar{t}_1)\delta t_1 + \bar{F}_1 \red{\overbrace{ [ x(t_0) - \bar{x}(t_0) ]\delta t_1 }^{\square}} + [ \bar{F}_1 A_1 x(t_0) + A_1 \bar{G}_1 + B_1 ] \purple{\overbrace{\delta t_1\delta t_1}^{\bigcirc}} 
	\nonumber
\end{equation}

\noindent 
both $\red{\square}$ and $\purple{\bigcirc}$ are products of differences that produces a smaller number and they will be neglected in the first order aproximations

\noindent
\begin{equation}
	\label{eq:cap4:linearization-x1-delta-t1-approximation}
	x(t_1)\delta t_1 \doteq \bar{x}(\bar{t}_1)\delta t_1 
\end{equation}

\noindent 
replacing \eqref{eq:cap4:linearization-error-x1-x0-result} and \eqref{eq:cap4:linearization-x2-reference-triangle-term} in \eqref{eq:cap4:linearization-x2-x1-recurrence} it follows that

\noindent
\begin{equation}
	\begin{split}
		x(t_2) - \bar{x}(\bar{t}_2) =  & \bar{F}_2 \{\green{
										\bar{F}_1 [x(t_0) - \bar{x}(t_0) + [\bar{F}_1 A_1 x(t_0) + A_1 \bar{G}_1 + B_1] \delta t_1]
								 }\} \\
							     & \quad - \bar{F}_2 [A_2 \blue{\bar{x}(t_1)} + B_2] \delta t_1 \\
							     & \quad + [ \bar{F}_2 A_2 \blue{\bar{x}(t_1)} + A_2 \bar{G}_2 + B_2] \delta t_2
	\end{split}
	\nonumber
\end{equation}

\noindent
\boxeq{eq:cap4:linearization-error-x2-x1-result}
{
	\begin{split}
		x(t_2) - \bar{x}(\bar{t}_2) =  & \bar{F}_2 \bar{F}_1 [x(t_0) - \bar{x}(t_0)] \\
							     & \quad + \bar{F}_2 [\bar{F}_1 A_1 x(t_0) + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2] \delta t_1 \\
							     & \quad + [\bar{F}_2 A_2 \bar{x}(\bar{t}_1) + A_2 \bar{G}_2 + B_2] \delta t_2
	\end{split}
}
