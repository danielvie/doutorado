dinamica a tempo discreto

\begin{equation}
	x(k + 1) = Ax(k) + Bu(k)
\end{equation}

com entrada u(k) $\in \mathbb{R}^p$ e estado $x(k) \in \mathbb{R}$ sujeitos aas seguintes restricoes:

\begin{equation}
	u(k) \in \mathcal{U}, x(k) \in \mathcal{X}, \forall k \geq 0
\end{equation}

\begin{equation}
	\mathcal{U} = { u \in \mathbb{r}^p : S_u u \leq b_u }, \mathcal{X} = { x \in \mathbb{R}^n : S_x x \leq b_x}
\end{equation}

deseja-se conduzir o estado para um ponto de equilibrio $\bar{x} \in int(\mathcal{X})$, associado a um valor de equilibrio $\bar{u} \in int(\mathcal{U})$.

Suponha que, sob a acao de uma dada lei de controle estabilizante da forma

\begin{equation}
	u(k) = -K [x(k) - \bar{x}] + \bar{u}
\end{equation}

o maximo conjunto admissivel quanto aas restricoes de estado e controle seja descrito por

\begin{equation}
	\mathcal{X}_f = \{ x \in \mathbb{r}^n : S_f x \leq b_f \}
\end{equation}

Se o estado inicial $x(0)$ estiver em $\mathcal{X}_f$, a lei de controle (*) podera ser aplicada a partir de $k = 0$ e o estado sera conduzido ao ponto de 
equilibrio desejado, sem que as restricoes sejam violadas

\fbox{\begin{minipage}{15em}
	O que fazer se $x(0) \notin \mathcal{X}_f$ ?
\end{minipage}}

CONDUCAO DO ESTADO AO CONJUNTO $\mathcal{X}_f$

Dado um estado inicial $x(0) \notin \mathcal{X}_f$ gostariamos de obter uma sequencia de $N$ acoes de controle de modo que

\begin{equation}
	\begin{split}
		u(0) \in \mathcal{U} \\
		u(1) \in \mathcal{U} \\
		\vdots \\
		u(N - 1) \in \mathcal{U}
	\end{split}
\end{equation}

de modo que 

\begin{equation}
	\begin{split}
		u(1) \in \mathcal{X} \\
		u(2) \in \mathcal{X} \\
		\vdots \\
		u(N - 1) \in \mathcal{X} \\
		u(N) \in \mathcal{X}
	\end{split}
\end{equation}

de maneira geral, a solucao (se existir) nao sera unica. portanto, eh necessario adotar um criterio que permita escolher a 
melhor sequencia de acoes de controle para realizar a tarefa

para isso, empregaremos um custo da forma

\begin{equation}
	J = \blue{\Vert x(N) -\bar{x} \Vert^2_{P_f}} + \red{\sum^{N-1}_{k=0} \Big[ \Vert x(k) - \bar{x} \Vert^2_Q +  \Vert u(k) - \bar{u} \Vert^2_R \Big]}
\end{equation}

sendo $P_f$, $Q$, $R$ matrizes de peso simetricas e positivo-definidas

A expressao de J envolve uma parcela de \blue{custo terminal} e outra de \red{custo de trejetoria}

Consideremos inicialmente $\bar{x} = 0$ e $\bar{u} = 0$.

\begin{equation}
	J = \Vert x(N) \Vert^2_{P_f} + \sum^{N-1}_{k=0} \Big[ \Vert x(k) \Vert^2_Q +  \Vert u(k) \Vert^2_R \Big]
\end{equation}

a expressao para o custo pose ser reescrita empregando a seguinte notacao:

\begin{equation}
	\mathbf{u} = 
	\left[
		\begin{matrix}
			u(0) \\		
			u(1) \\		
			\vdots \\
			u(N-1)
		\end{matrix}
	\right]
	\in \mathbb{R}^{pN},
	\mathbf{x} = 
	\left[
		\begin{matrix}
			x(1) \\ 
			x(2) \\ 
			\vdots \\
			x(N)
		\end{matrix}
	\right]
	\in \mathbb{R}^{nN}
\end{equation}

lembrando que o estado inicial $x(0)$ foi fixado na formulacao do problema.

com isso, a somatoria

\begin{equation}
	\sum^{N-1}_{k=0} \Vert u(k) \Vert^2_R = u^T(0) R u(0) + u^t(1)Ru(1) + \dotsm + u^T (N-1) R u(N-1)
\end{equation}

fica na forma

\begin{equation}
	\underbrace{
		\left[
			\begin{matrix}
				u^T(0) & u^T(1) & \dotsi & u^T(N - 1) 
			\end{matrix}
		\right]
	}_{\mathbf{u}^T}
	\left[
		\begin{matrix}
			R & 0 & \dotsi & 0 \\
			0 & R & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & R \\
		\end{matrix}
	\right]
	\underbrace{
		\left[
			\begin{matrix}
				u(0) \\
				u(1) \\
				\vdots \\
				u(N - 1)
			\end{matrix}
		\right]
	}_{\mathbf{u}}
\end{equation}

tendo-se omitido a dimensao ($p \times p$) das submatrizes de zeros, por brevidade.

Conventionando ainda a notacao

\begin{equation}
	\mathbf{R} = \left[
		\begin{matrix}
			R & 0 & \dotsi & 0 \\
			0 & R & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & R 
		\end{matrix}
	\right]
\end{equation}

ou, de forma abreviada:

\begin{equation}
	\mathbf{R} = diag_N(R)
\end{equation}

pode-se escrever

\begin{equation}
	\sum_{k=0}^{N-1} \Vert u(k) \Vert^2_R = \Vert \mathbf{u} \Vert^2_{\mathbf{R}}
\end{equation}

De forma similar, pode-se escrever

\begin{equation}
	\begin{gathered}
		\Vert x(N) \Vert^2_{P_f} + \sum_{k=0}^{N-1} \Vert x(k) \Vert^2_Q = x^T(0) Q x(0) \\
		+ x^T(1)Qx(1) + \dotsi + x^T(N-1)Qx(N-1) + x^T(N)P_fx(N) \\
		= x^T(0)Qx(0) +  \\
		\underbrace{
			\left[
				\begin{matrix}
					x^T(1) & x^T(2) & \dotsi & x^T(N)		
				\end{matrix}
			\right]
		}_{\mathbf{x}^T}
		\left[
			\begin{matrix}
				Q & 0 & \dotsi & 0 \\
				0 & Q & \dotsi & 0 \\
				\vdots & \vdots & \ddots & \vdots \\
				0 & 0 & \dotsi & P_f
			\end{matrix}
		\right]
		\underbrace{
			\left[
				\begin{matrix}
					x(1) \\
					x(2) \\
					\vdots \\
					x(N)
				\end{matrix}
			\right]
		}_{\mathbf{x}}
	\end{gathered}
\end{equation}

Adotando a notaçăo

\begin{equation}
	Q = 
	\left[
		\begin{matrix}
			Q & 0 & \dotsi & 0 \\
			0 & Q & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & P_f
		\end{matrix}
	\right]
	\in \mathbb{R}^{nN \times nN}
\end{equation}

ou, de forma abreviada:

\begin{equation}
	Q = 
	\left[
		\begin{matrix}
			diag_{N-1}(Q) & 0 \\
			0 & P_f
		\end{matrix}
	\right]
\end{equation}

pode-se escrever

\begin{equation}
	\Vert x(N) \Vert^2_{P_f} + \sum_{k=0}^{N-1} \Vert x(k) \Vert^2_Q + \Vert \mathbf{x} \Vert^2_{\mathbf{Q}}
\end{equation}

Em resumo, o custo pode ser reescrito como

\begin{equation}
	J(\mathbf{x}, \mathbf{u}) = \Vert x(0) \Vert^2_Q + \Vert \mathbf{x} \Vert^2_{\mathbf{Q}} + \Vert \mathbf{u} \Vert^2_{\mathbf{R}}
\end{equation}

Vamos agora expressar as restriçőes em termos de $\mathbf{x}$, $\mathbf{u}$.

As restriçőes podem ser escritas como

\begin{equation}
	\mathbf{u} = \left[
		\begin{matrix}
			u(0) \\
			u(1) \\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right]
	\in \mathcal{U} \times  \mathcal{U} \times \dotsi \times \mathcal{U} = \mathcal{U}^N
\end{equation}

\begin{equation}
	\mathbf{x} = \left[
		\begin{matrix}
			x(1) \\
			x(2) \\
			\vdots \\
			x(N)
		\end{matrix}
	\right]
	\in \mathcal{X} \times  \mathcal{X} \times \dotsi \times \mathcal{X} = \mathcal{X}^{N-1} \times \mathcal{X}_f
\end{equation}

Lembrando que

\begin{equation}
	\mathcal{U} = \left\{
		\mathbf{u} \in \mathbb{R}^p : S_u u \leq b_u	
	\right\}
\end{equation}

a restriçăo sobre $\mathbf{u}$ pode ser expressa como
\begin{equation}
	\left[
		\begin{matrix}
			S_u & 0 & \dotsi & 0 \\
			0 & S_u & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & S_u
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			u(0) \\
			u(1) \\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right]
	\leq
	\left[
		\begin{matrix}
			b_u \\
			b_u \\
			\vdots \\
			b_u
		\end{matrix}
	\right]
\end{equation}

ou, de forma mais concisa:

\begin{equation}
	\mathbf{S_u u} \leq \mathbf{b_u}
\end{equation}

sendo

\begin{equation}
	\mathbf{S_u} = diag_N(S_u), \mathbf{b_u} = col_N(b_u)
\end{equation}

De forma similar, lembrando que

\begin{equation}
	\mathcal{X} = \left\{
		x \in \mathbb{R}^n : S_x x \leq b_x
	\right\}
	, \mathcal{X}_f = 
	\left\{
		x \in \mathbb{R}^n : S_f x \leq b_f
	\right\}
\end{equation}

ou, de forma mais concisa

\begin{equation}
	\mathbf{S_x x} \leq \mathbf{b_x}
\end{equation}

sendo

\begin{equation}
	\mathbf{S_x} = 
	\left[
		\begin{matrix}
			diag_{N-1}(S_x) & 0 \\
			0 & S_f
		\end{matrix}
	\right], \mathbf{b_x} = 
	\left[
		\begin{matrix}
			col_{N-1}(b_x) \\
			b_f
		\end{matrix}
	\right]
\end{equation}

No caso geral em que $\bar{x} \ne 0, \bar{u} \ne 0$

\begin{equation}
	J = \Vert x(N) - \bar{x} \Vert^2_{P_f} + \sum_{k=0}^{N-1} \left[
		\norma{x(k) - \bar{x}}{Q} + \norma{u(k) - \bar{u}}{R}
	\right]
\end{equation}

torna-se

\begin{equation}
	J(\mathbf{x}, \mathbf{u}) = \norma{x(0) - \bar{x}}{P_f} + \norma{\mathbf{x}-col_N(\bar{x})}{\mathbf{Q}}
		+ \norma{\mathbf{u} - col_N(\bar{u})}{\mathbf{R}}
\end{equation}

O problema de otimizaçăo a ser resolvido pode ser expresso como

\begin{equation}
	\min_{\mathbf{x} \in \mathbb{R}^{nN}, \mathbf{u} \in \mathbb{R}^{pN}} J(x,u)
\end{equation}

sendo

\begin{equation}
	J(\mathbf{x}, \mathbf{u}) = \norma{x(0) - \bar{x}}{Q} + \norma{\mathbf{x} - col_N(\bar{x})}{\mathbf{Q}} + \norma{\mathbf{u} - col_N(\bar{u})}{\mathbf{R}}
\end{equation}

com $\mathbf{u}$, $\mathbf{u}$ sujeitas a

\begin{equation}
	\mathbf{S_x x} \leq \mathbf{b_x}
\end{equation}

\begin{equation}
	\mathbf{S_u u} \leq \mathbf{b_u}
\end{equation}

Vamos expressar o vinculo entre os vetores

\begin{equation}
	\mathbf{u} = \left[
		\begin{matrix}
			u(0) \\ 
			u(1) \\ 
			\vdots \\ 
			u(N-1)
		\end{matrix}
	\right],
	\mathbf{x} = \left[
		\begin{matrix}
			x(1) \\ 
			x(2) \\ 
			\vdots \\ 
			x(N)
		\end{matrix}
	\right]
\end{equation}

com base na equaçăo de estado

\begin{equation}
	x(k+1) = Ax(k) + Bu(k)
\end{equation}

e na condiçăo inicial $x(0)$.

Neste ponto, vale lembrar a soluçăo da equaçăo de estado:

\begin{equation}
	x(k) = A^kx(0) + \sum_{i=0}^{k-1} A^{k-1-i} B u(i)
\end{equation}

que pose ser escrita para $k = 1,2,...,N$ como

\begin{align}
	x(1) &= A x(0) + Bu(0) \\
	x(2) &= A^2 x(0) + ABu(0) + Bu(1) \\
	&\vdots \\
	x(N) &= A^N x(0) A^{N - 1}Bu(0) + \dotsi + ABu(N - 2) + Bu(N - 1)
\end{align}

ou ainda

\begin{equation}
	\left[
		\begin{matrix}
			x(1) \\
			x(2) \\
			\vdots \\
			x(N) \\
		\end{matrix}
	\right] = 
	\left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	x(0) + 
	\left[
		\begin{matrix}
			B & 0 & \dotsi & 0 \\
			AB & B & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \dotsi & B
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			u(0) \\
			u(1)\\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right]
\end{equation}

De forma mais concisa, vamos escrever

\begin{equation}
	\mathbf{x} = \mathbf{A}x(0) + \mathbf{Bu}
\end{equation}

com 

\begin{equation}
	\mathbf{A} = \left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	, 
	\mathbf{B} = \left[
		\begin{matrix}
			B & 0 & \dotsi & 0 \\
			AB & B & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \dotsi & B
		\end{matrix}
	\right]
\end{equation}

Tendo em vista o vínculo

\begin{equation}
	\mathbf{x} = \mathbf{A}x(0) + \mathbf{Bu}
\end{equation}

pode-se reescrever a funçăo de custo como

\begin{equation}
	\begin{gathered}
		J(\mathbf{x}, \mathbf{u}) = \norma{x(0) - \bar{x}}{Q} + \norma{\mathbf{x} - col_N(\bar{x})}{\mathbf{Q}} + 
		\norma{\mathbf{u} - col_N(\bar{u})}{\mathbf{R}} \\
		= \norma{x(0) - \bar{x}}{Q} \\
		+ \left[ \mathbf{A} x(0) + \mathbf{Bu} - col_N(\bar{x})\right]^T \mathbf{Q}
		\left[ \mathbf{A} x(0) + \mathbf{Bu} - col_N(\bar{x})\right] \\
		+ \left[\mathbf{u} - col_N(\bar{u})\right]^T \mathbf{R}
		\left[\mathbf{u} - col_N(\bar{u})\right] \\
		J(\mathbf{x}, \mathbf{u}) = \norma{x(0) - \bar{x}}{Q} \\
		+ \left[\mathbf{A}x(0) - col_N(\bar{x}) \mathbf{Bu}\right]^T \mathbf{Q}
		\left[\mathbf{A}x(0) - col_N(\bar{x}) \mathbf{Bu}\right] \\
		+ \mathbf{u}^T \mathbf{Ru} - \mathbf{u}^T\mathbf{R} \left[col_N(\bar{u})\right]
		- \left[col_N(\bar{u}) \right]^T \mathbf{Ru} + \norma{col_N(\bar{u})}{\mathbf{R}}
	\end{gathered}
\end{equation}

\begin{equation}
	\begin{gathered}
		J(\mathbf{x}, \mathbf{u}) = 
		\blue{ \norma{x(0) - \bar{x}}{Q} +  + \norma{col_N(\bar{u})}{\mathbf{R}} } \\
		\green{ \left[\mathbf{A}x(0) - col_N(\bar{x})\right]^T \mathbf{QBu} + y} \\
		-\mathbf{u}^T\mathbf{R} \left[col_N(\bar{u})\right] - \left[col_N(\bar{u})\right]^T \mathbf{Ru} + \mathbf{u}^T (\mathbf{R} + \mathbf{B}^T \mathbf{QB})\mathbf{u}
	\end{gathered}
\end{equation}

Observacoes:

O \blue{termo constante} será denotado por \blue{$c_0$}

Lembrando que a matriz $\mathbf{Q}$ é simétrica, tem-se

\begin{equation}
	\green{ \left[\mathbf{A}x(0) - col_N(\bar{x})\right]^T \mathbf{QBu} } 
	= 
	\left\{
	\green{ \mathbf{u}^T \mathbf{B}^T \mathbf{Q} \left[\mathbf{A}x(0) - col_N(\bar{x})\right] }
	\right\}^T \\
\end{equation}

Como esses termos săo escalares, conclui-se que săo iguais entre si. Logo:

\begin{equation}
	\begin{gathered}
		\green{ \left[\mathbf{A}x(0) - col_N(\bar{x})\right]^T \mathbf{QBu} }
		+ \green{ \mathbf{u}^T \mathbf{B}^T \mathbf{Q} \left[\mathbf{A}x(0) - col_N(\bar{x})\right] } \\
		= \green{ 2 \left[\mathbf{A}x(0) - col_N(\bar{x})\right]^T \mathbf{QBu} }
	\end{gathered}
\end{equation}

De forma similar:

\begin{equation}
	- \mathbf{u}^T \mathbf{R} \left[col_N(\bar{u})\right] - \left[col_N(\bar{u})\right]^T \mathbf{Ru} = 
	-2 \left[col_N(\bar{u})\right]^T \mathbf{Ru}
\end{equation}

Finalmente, pode-se escrever

\begin{equation}
	\begin{gathered}
		J(\mathbf{u}) = \mathbf{u}^T (\mathbf{B}^T \mathbf{Q} \mathbf{B} + \mathbf{R}) \mathbf{u} \\
		+ 2 \left\{
			\left[
				\mathbf{A}x(0) - col_N(\bar{x})
			\right]^T \mathbf{QB} - \left[col_N(\bar{u})\right]^T \mathbf{R}
		\right\} \mathbf{u} = c_0
	\end{gathered}
\end{equation}

com 

\begin{equation}
	c_0 = \norma{x(0) - \bar{x}}{Q} + \norma{\mathbf{A}x(0) - col_N(\bar{x})}{\mathbf{R}} + \norma{col_N(\bar{u})}{\mathbf{R}}
\end{equation}

Note-se que passaremos a usar a notaçăo $J(\mathbf{u})$ em lugar de $J(\mathbf{x}, \mathbf{u})$, uma vez que $\mathbf{x}$ foi eliminada da expressăo do custo.

Para terminar a reformulaçăo do problema de otimizaçăo em termos de $\mathbf{u}$, deve-se considerar o vínculo entre $\mathbf{x}$ e $\mathbf{u}$ nas \textbf{restriçőes} 

Considerando novamente o vínculo

\begin{equation}
	\mathbf{x} = \mathbf{A} x(0) + \mathbf{Bu}
\end{equation}

a restriçăo 

\begin{equation}
	\mathbf{S_x x} \leq \mathbf{b_x}
\end{equation}

pode ser reformulada como

\begin{equation}
	\mathbf{S_x B u} \leq \mathbf{b_x} - \mathbf{S_x A} x(0)
\end{equation}

Em conjunto com $\mathbf{S_u u} \leq \mathbf{b_u}$, pode-se entăo expressar as restriçőes na forma

\begin{equation}
	\left[
		\begin{matrix}
			\mathbf{S_xB} \\
			\mathbf{S_u}
		\end{matrix}
	\right] \mathbf{u} \leq
	\left[
		\begin{matrix}
			\mathbf{b_x} - \mathbf{S_x A} x(0) \\
			\mathbf{b_u}
		\end{matrix}
	\right]
\end{equation}

ou, de maneira mais sucinta:

\begin{equation}
	\mathbf{Su} \leq \mathbf{b}
\end{equation}

com

\begin{equation}
	\mathbf{S} = \left[
		\begin{matrix}
			\mathbf{S_xB} \\
			\mathbf{S_u}
		\end{matrix}
	\right],
	\mathbf{b} = \left[
		\begin{matrix}
			\mathbf{b_x} - \mathbf{S_x A} x(0) \\
			\mathbf{b_u}
		\end{matrix}
	\right]
\end{equation}

Em resumo, o problema de otimizaçăo passa a ser expresso como

\begin{equation}
	\min_{\mathbf{u} \in^{pN}} J(\mathbf{u}) s.a. \mathbf{Su} \leq \mathbf{b}
\end{equation}

sendo

\begin{equation}
	\begin{gathered}
		J(\mathbf{u}) \mathbf{u^T} (\mathbf{B}^T \mathbf{Q} \mathbf{B} + \mathbf{R})\mathbf{u} \\
		+ 2 \Big\{
			[\mathbf{A} x(0) - col_N(\bar{x})]^T \mathbf{Q} \mathbf{B} - [col_N(\bar{u})]^T \mathbf{R}
		\Big\} \mathbf{u} + c_0
	\end{gathered}
\end{equation}

Trata-se da minimizaçăo de um \textbf{custo quadrático} sujeito a \textbf{restriçőes lineares}:

\textbf{PROBLEMA DE PROGRAMACAO QUADATICA}

Para facilitar a análise do problema de programaçăo quadrática, passaremos a escrever o custo

\begin{equation}
	\begin{gathered}
		J(\mathbf{u}) = \mathbf{u}^T (\mathbf{B}^T \mathbf{Q} \mathbf{B} + \mathbf{R}) \mathbf{u} + \\
		2 \Big\{
			[\mathbf{A} x(0) - col_N(\bar{x})]^T \mathbf{Q} \mathbf{B} - [col_N(\bar{u})]^T \mathbf{R}
		\Big\} \mathbf{u} + c_0
	\end{gathered}
\end{equation}

nesta forma padronizada:

\begin{equation}
	J(\mathbf{u}) = \frac{1}{2} \mathbf{u}^T J_{\boldmath{uu}}\mathbf{u} f^T \mathbf{u} + c_0
\end{equation}

com 

\begin{equation}
	J(\mathbf{u}) = 2 (\mathbf{B}^T \mathbf{Q} \mathbf{B} + \mathbf{R}), f = 2 
	\Big\{
		\mathbf{B}^T \mathbf{Q} [ \mathbf{A} x(0) - col_N(\bar{x}) ] - \mathbf{R} col_N(\bar{u})
	\Big\}
\end{equation}

lembrando que $\mathbf{Q}$ e $\mathbf{R}$ săo simétricas.

TERMINOLOGIA E NOTACAO

\begin{equation}
	J(\mathbf{u}) \frac{1}{2} \mathbf{u}^T J_{\mathbf{uu}u} + f^T\mathbf{u} = c_0
\end{equation}

1) A matriz $J_{\mathbf{uu}}$ corresponde ŕ chamada \textbf{Hessiana} do custo, que é definica como

\begin{equation}
	J_{\mathbf{uu}} = 
	\left[
		\begin{matrix}
			\frac{\partial^2 J}{\partial u^2_1} & \frac{\partial^2 J}{\partial u_1 \partial u_2} & \dotsi & \frac{\partial^2 J}{\partial u_1 \partial u_N} \\ 
			\frac{\partial^2 J}{\partial u_2 \partial u_1} & \frac{\partial^2 J}{\partial u^2_2} & \dotsi & \frac{\partial^2 J}{\partial u_2 \partial u_N} \\ 
			\vdots & \vdots & \ddots & \vdots \\ 
			\frac{\partial^2 J}{\partial u_N \partial u_1} & \frac{\partial^2 J}{\partial u_N \partial u_2} & \dotsi & \frac{\partial^2 J}{\partial u^2_N} \\ 
		\end{matrix}
	\right]
\end{equation}

sendo $u_1, u_2, \dotso, u_N$ como componentes de $\mathbf{u}$. Vide \textbf{material suplementar 1} ao final dos slides para mais detalhes.

Vale notar que $J_{ \mathbf{uu} } = 2(\mathbf{B}^T \mathbf{Q} \mathbf{B} + \mathbf{R})$ é simétrica e positivo definida, pois $\mathbf{Q}$ e $\mathbf{R}$ 
săo simétricas e positivo definidas.
