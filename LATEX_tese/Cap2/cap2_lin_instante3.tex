Solucao no dominio do tempo para o instante de chaveamento 3


\begin{align}
	\label{eq_cap2_lin_x3}
	x(t_3)  &= e^{A_2(t_3-t_2)}x(t_2) + \intlinAA{2}{3}{}{} \\
	\label{eq_cap2_lin_xr3}
	x(t_{r_3}) &= e^{A_2(t_{r_3}-t_{r_2})}x(t_{r_2}) + \intlinA{2}{3}{r}{r}
\end{align}

subtraindo \ref{eq_cap2_lin_x3} e \ref{eq_cap2_lin_xr3} temos

\begin{equation}
	\label{eq_cap2_lin_erro_x2x3}
	\begin{split}
		x(t_3) - x(t_{r_3}) = \green{e^{A_2(t_3-t_2)}x(t_2) - e^{A_2(t_{r_3}-t_{r_2})}x(t_{r_2})} \\ 
		+ \underbrace{\orange{\intlinAA{2}{3}{}{} - \intlinA{2}{3}{r}{r}}}_{teste}
	\end{split}
\end{equation}

Desenvolvendo o destaque em \green{verde}

% 2.16
\begin{align}
	\green{ \square } &=
	\green{
		e^{A_2(t_{r_3} + \delta t_3 - t_{r_2} - \delta t_2)} x(t_2)
		- e^{A_2(t_{r_3} - t_{r_2})} x_r(t_{r_2})
	} \nonumber \\
	&= \green{
		\blue{\Fri{2}{3}} \purple{e^{A_2(\delta t_3 - \delta t_2)}} x(t_2)
		- \blue{\Fri{2}{3}} x_r(t_{r_2})
	} \nonumber \\
	&= \green{
		\blue{F_{r_2}} \purple{[I + A_2(\delta t_3 - \delta t_2)]} x(t_2)
		- \blue{F_{r_2}} x_r(t_{r_2})
	} \nonumber \\
	\label{eq_cap2_lin_x3_green}
	&= \green{
		F_{r_2} [x(t_2) - x_r(t_{r_2})] + F_{r_2} A_2 x(t_2) \delta t_3
		- F_{r_2} A_2 x(t_2) \delta t_2
	}
\end{align}

Desenvolvendo o destaque em \orange{laranja}

% 2.17
\begin{align}
	\orange{ \square } 	&= \orange{\intlinAA{2}{3}{}{} - \intlinA{2}{3}{r}{r}} \nonumber \\
					   	&= \orange{ \intlinC{t_{r_2}+\delta t_2}{t_{r_3}+\delta t_3}{2}{3} - \intlinA{2}{3}{r}{r}} \nonumber \\
						\begin{split}
							&= \orange{ \purple{ e^{A_2 \delta t_3} }
								\bigg[
									\blue{\intlinA{2}{3}{r}{r}} - \intlinB{t_{r_2}}{t_{r_2} + \delta t_2}{2}{3} + \intlinB{t_{r_3}}{t_{r_3} + \delta t_3}{2}{3} \bigg] } \\
							& \qquad \orange{ - \blue{\intlinA{2}{3}{r}{r}} }
						\end{split} \nonumber \\
						&= \orange{ \purple{(I + A_2 \delta t_3)}
							\bigg[
								\blue{G_{r_2}} - \pink{\intlinB{t_{r_2}}{t_{r_2} + \delta t_2}{2}{3}} + \brown{\intlinB{t_{r_3}}{t_{r_3} + \delta t_3}{2}{3}}
							\bigg] - \blue{G_{r_2}}
						} \nonumber \\
						&= \orange{ (I + A_2 \delta t_3)
							\bigg[
								G_{r_2} - \pink{e^{A_2(t_{r_3} - t_{r_2})} B_2 \delta t_2} + \brown{B_2 \delta t_3}
							\bigg] - G_{r_2}
						} \nonumber \\
						&= \orange{ (I + A_2 \delta t_3)
							\bigg[
								G_{r_2} - \pink{F_{r_2} B_2 \delta t_2} + B_2 \delta t_3
							\bigg] - G_{r_2}
						} \nonumber \\
						&= \orange{ \cancel{G_{r_2}} - F_{r_2} B_2 \delta t_2 + B_2 \delta t_3 + A_2 G_{r_2} \delta t_3 - \cancel{G_{r_2}} } \nonumber \\
						\label{eq_cap2_lin_x3_orange}
						&= \orange{ A_2 G_{r_2} \delta t_3 + B_2 \delta t_3 - F_{r_2} B_2 \delta t_2 }
\end{align}

substituindo (\ref{eq_cap2_lin_x3_green}) e (\ref{eq_cap2_lin_x3_orange}) de volta na equaçăo (\ref{eq_cap2_lin_erro_x2x3})


\begin{align}
	\begin{split}
		x(t_3) - x_r(t_{r_3}) =  & F_{r_2} [\green{x(t_2) - x_{r_2}(t_{r_2})}] \\
							     & - F_{r_2} [A_2 \blue{x(t_2)} + B_2] \delta t_2 \\
							     & + [ F_{r_2} A_2 \blue{x(t_2)} + A_2 G_{r_2} + B_2] \delta t_3
	\end{split}
\end{align}

\begin{align}
	\begin{split}
		x(t_3) - x_r(t_{r_3}) =  & F_{r_2} \{\green{F_{r_1}F_{r_0}[x(t_0) - x_r(t_0)]} \\
								 & \green{ + F_{r_1}[F_{r_0} A_0 x(t_0) + A_0 G_{r_0} - A_1 x_r(t_{r_1}) + B_0 - B_1] \delta t_1 } \\
								 & \green{ + [F_{r_1} A_1 x_r(t_1) + A_1 G_{r_1} + B_1 ] \delta t_2 } \} \\
							     & - F_{r_2} [A_2 \blue{x_r(t_2)} + B_2] \delta t_2 \\
							     & + [ F_{r_2} A_2 \blue{x_r(t_2)} + A_2 G_{r_2} + B_2] \delta t_3
	\end{split}
\end{align}

\begin{align}
	\begin{split}
		x(t_3) - x_r(t_{r_3}) =  & F_{r_2}F_{r_1}F_{r_0} [x(t_0) - x_r(t_0)] \\
								 & + F_{r_2}F_{r_1}[F_{r_0} A_0 x(t_0) + A_0 G_{r_0} - A_1 x_r(t_{r_1}) + B_0 - B_1] \delta t_1 \\
								 & + F_{r_2}[F_{r_1} A_1 x_r(t_1) + A_1 G_{r_1} - A_2 x_r(t_{r_2}) + B_1 - B_2 ] \delta t_2 \\
							     & + [F_{r_2} A_2 x_r(t_2) + A_2 G_{r_2} + B_2] \delta t_3
	\end{split}
\end{align}
