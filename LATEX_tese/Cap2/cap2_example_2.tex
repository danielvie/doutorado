\subsection{Multilevel converter}
\label{cap2-session-patino2}

The second application example is also extracted from \citeonline{patino2010predictive}, 
a Multilevel converter with topology is shown in Figure \refeq{cap2-circuit-ex-2}. 

\begin{figure}[H]
    \centering
    \includegraphics[width=10cm]{Cap2/fig/figura_circuito_ex_2.eps} 
    \caption{Four-level three-cell converter}
    \label{cap2-circuit-ex-2}
\end{figure}

The state vector is the voltage on the capacitors \highlight{$x(t) = [v_{c_1}(t), v_{c_2}(t), i_L(t)]$}, 
where \highlight{$v_{c_1}$}, \highlight{$v_{c_2}$} are the voltages in each capacitor 
and \highlight{$i_L$} is the load current.

The dynamic of this system can be described as

\noindent
\begin{equation}
    \left[ \begin{array}{c}
        \dot{x}_1(t) \\
        \dot{x}_2(t) \\
        \dot{x}_3(t)
    \end{array} \right]
    =
    \left[ \begin{array}{ccc}
        -\frac{x_3(t)}{C_1} & \frac{x_3(t)}{C_1} & 0 \\
        0 & -\frac{x_3(t)}{C_2} & \frac{x_3(t)}{C_2} \\
        \frac{x_1(t)}{L} & \frac{x_2(t) - x_1(t)}{L} & \frac{E - x_2(t)}{L}
    \end{array} \right]
    \left[ \begin{array}{c}
        u_1(t) \\
        u_2(t) \\
        u_3(t)
    \end{array} \right]
    +
    \left[ \begin{array}{c}
        0 \\
        0 \\
        -\frac{R}{L}x_3(t)
    \end{array} \right]
\end{equation}

\noindent
rearranging the terms we have

\noindent
\begin{equation}
    \left[ \begin{array}{c}
        \dot{x}_1(t) \\
        \dot{x}_2(t) \\
        \dot{x}_3(t)
    \end{array} \right]
    =
    \left[ \begin{array}{ccc}
        0 & 0 & \frac{u_2 - u_1}{C1} \\
        0 & 0 & \frac{u_3 - u_2}{C2} \\
        \frac{u_1-u_2}{L} & \frac{u_2-u_3}{L} & \frac{-R}{L}
    \end{array} \right]
    \left[ \begin{array}{c}
        x_1(t) \\
        x_2(t) \\
        x_3(t)
    \end{array} \right]
    +
    \left[ \begin{array}{c}
        0 \\
        0 \\
        -\frac{R}{L}u_3
    \end{array} \right]
\end{equation}

Each combination of the switches determines a different behavior:

\noindent
\begin{align}
    % 1 
    \mathbf{u_1} &= \left[ \begin{array}{ccc} 0 \\ 0 \\ 0 \end{array} \right],
    A_1 = \left[ \begin{array}{ccc} 
        0 & 0 & 0 \\
        0 & 0 & 0 \\
        0 & 0 & \frac{-R}{L} 
    \end{array} \right],
    b_1 = \left[ \begin{array}{c} 0 \\ 0 \\ 0 \end{array} \right] \\
    % 2
    \mathbf{u_2} &= \left[ \begin{array}{ccc} 0 \\ 0 \\ 1 \end{array} \right],
    A_2 = \left[ \begin{array}{ccc} 
        0 & 0 & 0 \\
        0 & 0 & \frac{1}{C_1} \\
        0 & \frac{-1}{L} & \frac{-R}{L}
    \end{array} \right],
    b_2 = \left[ \begin{array}{c} 0 \\ 0 \\ \frac{E}{L} \end{array} \right] \\
    % 3 
    \mathbf{u_3} &= \left[ \begin{array}{ccc} 0 \\ 1 \\ 0 \end{array} \right],
    A_3 = \left[ \begin{array}{ccc} 
        0 & 0 & \frac{1}{C_1} \\
        0 & 0 & \frac{-1}{C_2} \\
        \frac{-1}{L} & \frac{1}{L} & \frac{-R}{L}
     \end{array} \right],
    b_3 = \left[ \begin{array}{c} 0 \\ 0 \\ 0 \end{array} \right] \\
    % 4 
    \mathbf{u_4} &= \left[ \begin{array}{ccc} 0 \\ 1 \\ 1 \end{array} \right],
    A_4 = \left[ \begin{array}{ccc} 
        0 & 0 & \frac{1}{C_1} \\ 
        0 & 0 & 0 \\ 
        \frac{-1}{L} & 0 & \frac{-R}{L} 
    \end{array} \right],
    b_4 = \left[ \begin{array}{c} 0 \\ 0 \\ \frac{E}{L} \end{array} \right] \\
    % 5 
    \mathbf{u_5} &= \left[ \begin{array}{ccc} 1 \\ 0 \\ 0 \end{array} \right],
    A_5 = \left[ \begin{array}{ccc} 
         0 & 0 & \frac{-1}{C_1} \\
         0 & 0 & 0 \\
         \frac{1}{L} & 0 & \frac{-R}{L} 
    \end{array} \right],
    b_5 = \left[ \begin{array}{c} 0 \\ 0 \\ 0 \end{array} \right] \\
    % 6 
    \mathbf{u_6} &= \left[ \begin{array}{ccc} 1 \\ 0 \\ 1 \end{array} \right],
    A_6 = \left[ \begin{array}{ccc} 
         0 & 0 & \frac{-1}{C_1} \\
         0 & 0 & \frac{1}{C_2} \\
         \frac{1}{L} & \frac{-1}{L} & \frac{-R}{L} 
    \end{array} \right],
    b_6 = \left[ \begin{array}{c} 0 \\ 0 \\ \frac{E}{L} \end{array} \right] \\
    % 7 
    \mathbf{u_7} &= \left[ \begin{array}{ccc} 1 \\ 1 \\ 0 \end{array} \right],
    A_7 = \left[ \begin{array}{ccc} 
         0 & 0 & 0 \\
         0 & 0 & \frac{-1}{C_1} \\
         0 & \frac{1}{L} & \frac{-R}{L} 
    \end{array} \right],
    b_7 = \left[ \begin{array}{c} 0 \\ 0 \\ 0 \end{array} \right] \\
    % 8 
    \mathbf{u_8} &= \left[ \begin{array}{ccc} 1 \\ 1 \\ 1 \end{array} \right],
    A_8 = \left[ \begin{array}{ccc} 
         0 & 0 & 0 \\
         0 & 0 & 0 \\
         0 & 0 & \frac{-R}{L} 
    \end{array} \right],
    b_8 = \left[ \begin{array}{c} 0 \\ 0 \\ \frac{E}{L} \end{array} \right]
\end{align}

With the detailed dynamic for each switching position, we can define the list of 
elements for the multilevel converter as \highlight{$\Bigr\{ A_{1}, A_{2}, A_{3}, A_{4}, A_{5}, A_{6}, A_{7}, A_{8} \Bigr\}$} and 
\highlight{$\Bigr\{b_{1}, b_{2}, b_{3}, $} \highlight{$b_{4}, b_{5}, b_{6}, b_{7}, b_{8} \Bigr\}$}.


The criteria for the optimization is to compute the trajectory around the average value 
\highlight{$\Big[\frac{1}{3}E, \frac{2}{3}E, 1\Big]$} with \highlight{$E = 30 \; \mathrm{V}$}.
The maximum cycle period is \highlight{$T_{p,max} = 0.4 \; \mathrm{ms}$} and minimum tweel time of \highlight{$t_{min} = 0.022 \; \mathrm{ms}$}. 

\comentariopurple{
    unidades (como F, J, etc.) devem ser escritas em tipo romano, isto eh, nao italico.
    deve haver um espaco separando o valor da unidade.
}

The simulation parameters are \highlight{$C_1 = 40 \; \mathrm{\mu F}$}, 
\highlight{$C_2 = 40 \; \mathrm{\mu F}$}, \highlight{$L = 10 \; \mathrm{mH}$} and \highlight{$R = 10 \; \mathrm{\Omega}$} with the weight matrix \highlight{$Q = diag\big[10,5,20000\big]$}.

Using the optimal switching sequence
\begin{equation}
    \Omega^\infty = \Big\{ 1, 2, 4, 8, 3, 1, 5, 8, 5 \Big\} \\
\end{equation}

The optimal switching instant is

\comentarioblue{
    indicar unidades
}

\begin{equation}
    \tau^\infty = \Big\{ 0, 0.064, 0.088, 0.112, 0.134, 0.157, 0.206, 0.230, 0.253, 0.276 \Big\} \; \mathrm{ms}
\end{equation}

\noindent
with the initial condition \highlight{$x(0) = \Big[9.514 \; \mathrm{V}, 19.821 \; \mathrm{V},  1.031 \; \mathrm{A}\Big]^T$}.
