\subsection{Multilevel converter}

Considering a Multilevel converter whose topology is shown in Figure \refeq{cap2-circuit-ex-2}. 

\begin{figure}[H]
    \centering
    \includegraphics[width=10cm]{Cap2/fig/figura_circuito_ex_2.eps} 
    \caption{Four-level three-cell converter}
    \label{cap2-circuit-ex-2}
\end{figure}

The state variables are the voltage on the capacitors $x(t) = [v_{c_1}(t), v_{c_2}(t), i_L(t)]$. 
Where $v_{c_1}$, $v_{c_2}$ are the voltages in each capacitor and $i_L$ is the load current.

The dynamic of this system can be described as

\noindent
\begin{equation}
    \left[ \begin{array}{c}
        \dot{x}_1(t) \\
        \dot{x}_2(t) \\
        \dot{x}_3(t)
    \end{array} \right]
    =
    \left[ \begin{array}{ccc}
        -\frac{x_3(t)}{C_1} & \frac{x_3(t)}{C_1} & 0 \\
        0 & -\frac{x_3(t)}{C_2} & \frac{x_3(t)}{C_2} \\
        \frac{x_1(t)}{L} & \frac{x_2(t) - x_1(t)}{L} & \frac{E - x_2(t)}{L}
    \end{array} \right]
    \left[ \begin{array}{c}
        u_1(t) \\
        u_2(t) \\
        u_3(t)
    \end{array} \right]
    +
    \left[ \begin{array}{c}
        0 \\
        0 \\
        -\frac{R}{L}x_3(t)
    \end{array} \right]
\end{equation}

\begin{equation}
    A = 
    \left[
        \begin{array}{ccc}
            -\frac{x_3(t)}{C_1} & \frac{x_3(t)}{C_1} & 0 \\
            0 & -\frac{x_3(t)}{C_2} & \frac{x_3(t)}{C_2} \\
            \frac{x_1(t)}{L} & \frac{x_2(t) - x_1(t)}{L} & \frac{E-x_2(t)}{L}
        \end{array}
    \right]
\end{equation}

\noindent
which is submitted to the switch positions

\noindent
\begin{align}
    \mathbf{u_0} = \left[ \begin{array}{ccc} 0 & 0 & 0 \end{array} \right]^T \\
    \mathbf{u_1} = \left[ \begin{array}{ccc} 0 & 0 & 1 \end{array} \right]^T \\
    \mathbf{u_2} = \left[ \begin{array}{ccc} 0 & 1 & 0 \end{array} \right]^T \\
    \mathbf{u_3} = \left[ \begin{array}{ccc} 0 & 1 & 1 \end{array} \right]^T \\
    \mathbf{u_4} = \left[ \begin{array}{ccc} 1 & 0 & 0 \end{array} \right]^T \\
    \mathbf{u_5} = \left[ \begin{array}{ccc} 1 & 0 & 1 \end{array} \right]^T \\
    \mathbf{u_6} = \left[ \begin{array}{ccc} 1 & 1 & 0 \end{array} \right]^T \\
    \mathbf{u_7} = \left[ \begin{array}{ccc} 1 & 1 & 1 \end{array} \right]^T
\end{align}

Each combination of the switches determines a different behavior as shown next.


% "n
% \left[ \begin{array}{c}
%     0 \\
%     0 \\
%     -\frac{R}{L}x_3(t)
% \end{array} \right]

\begin{align}
    A \mathbf{u_0} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right] &= \left[ \begin{array}{c} 0 \\ 0 \\ \frac{-R x_3(t)}{L} \end{array} \right] = 
    \left[ \begin{array}{ccc} 0&0&0 \\ 0&0&0 \\ 0&0&-\frac{R}{L} \end{array} \right] 
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A \mathbf{u_1} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right]
    &= \left[ \begin{array}{c} 0 \\ \frac{x_3(t)}{C_2} \\ \frac{E-x_2(t)-R x_3(t)}{L} \end{array} \right] =
    \left[ \begin{array}{ccc} 0&0&0 \\ 0&0&\frac{1}{C_2} \\ 0&-\frac{1}{L}&-\frac{R}{L} \end{array} \right] 
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right] \\
    A \mathbf{u_2} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right]
    &= \left[ \begin{array}{c} \frac{x_3(t)}{C_1} \\ \frac{-x_3(t)}{C_2} \\ \frac{x_2(t)-x_1(t)-R x_3(t)}{L} \end{array} \right] =
    \left[ \begin{array}{ccc} 0&0&\frac{1}{C_1} \\ 0&0&-\frac{1}{C_2} \\ -\frac{1}{L}&\frac{1}{L}&-\frac{R}{L} \end{array} \right]
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A \mathbf{u_3} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right]
    &= \left[ \begin{array}{c} \frac{x_3(t)}{C_1} \\ 0 \\ \frac{E-x_1(t)-R x_3(t)}{L} \end{array} \right] =
    \left[ \begin{array}{ccc} 0 & 0 & \frac{1}{C_1} \\ 0 & 0 & 0 \\ -\frac{1}{L} & 0 &-\frac{R}{L} \end{array} \right]
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right] \\
    A \mathbf{u_4} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right]
    &= \left[ \begin{array}{c} \frac{-x_3(t)}{C_1} \\ 0 \\ \frac{x_1(t)-R x_3(t)}{L} \end{array} \right] =
    \left[ \begin{array}{ccc} 0&0&-\frac{1}{C_1} \\ 0&0&0 \\ \frac{1}{L}&0&-\frac{R}{L} \end{array} \right]
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A \mathbf{u_5} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right]
    &= \left[ \begin{array}{c} \frac{-x_3(t)}{C_1} \\ \frac{x_3(t)}{C_2} \\ \frac{E-x_1(t)-x_2(t)-R x_3(t)}{L} \end{array} \right] =
    \left[ \begin{array}{ccc} 0&0&-\frac{1}{C_1} \\ 0&0&\frac{1}{C_2} \\ \frac{1}{L}&-\frac{1}{L}&-\frac{R}{L} \end{array} \right]
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right] \\
    A \mathbf{u_6} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right]
    &= \left[ \begin{array}{c} 0 \\ \frac{-x_3(t)}{C_2} \\ \frac{x_2(t)-R x_3(t)}{L} \end{array} \right] =
    \left[ \begin{array}{ccc} 0&0&0\\ 0&0&-\frac{1}{C_2} \\ 0&\frac{1}{L}&-\frac{R}{L} \end{array} \right]
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A \mathbf{u_7} +
    \left[ \begin{array}{c} 0 \\ 0 \\ -\frac{R}{L}x_3(t) \end{array} \right]
    &= \left[ \begin{array}{c} 0 \\ 0 \\ \frac{E-R x_3(t)}{L} \end{array} \right] =
    \left[ \begin{array}{ccc} 0&0&0\\ 0&0&0\\ 0&0&-\frac{R}{L} \end{array} \right]
    \left[ \begin{array}{c} x_1(t)\\ x_2(t)\\ x_3(t) \end{array} \right] +
    \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right]
\end{align}

With the detailed dynamic for each switching positions we can define the list of elements for the multilevel converter as 
$\Bigr\{ A_{1}, A_{2}, A_{3}, A_{4}, A_{5}, A_{6}, A_{7} \Bigr\}$ and $\Bigr\{ b_{1}, b_{2}, b_{3}, b_{4}, b_{5}, b_{6}, b_{7} \Bigr\}$
with

\begin{align}
    A_{0} &= \left[ \begin{array}{ccc} 0&0&0 \\ 0&0&0 \\ 0&0&-\frac{R}{L} \end{array} \right],
    b_{0} = \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A_{1} &= \left[ \begin{array}{ccc} 0&0&0 \\ 0&0&\frac{1}{C_2} \\ 0&-\frac{1}{L}&-\frac{R}{L} \end{array} \right],
    b_{1} = \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right] \\
    A_{2} &= \left[ \begin{array}{ccc} 0&0&\frac{1}{C_1} \\ 0&0&-\frac{1}{C_2} \\ -\frac{1}{L}&\frac{1}{L}&-\frac{R}{L} \end{array} \right],
    b_{2} = \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A_{3} &= \left[ \begin{array}{ccc} 0&0&\frac{1}{C_1} \\ 0&0&0 \\ -\frac{1}{L}&0&-\frac{R}{L} \end{array} \right],
    b_{3} = \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right] \\
    A_{4} &= \left[ \begin{array}{ccc} 0&0&-\frac{1}{C_1} \\ 0&0&0 \\ \frac{1}{L}&0&-\frac{R}{L} \end{array} \right],
    b_{4} = \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A_{5} &= \left[ \begin{array}{ccc} 0&0&-\frac{1}{C_1} \\ 0&0&\frac{1}{C_2} \\ \frac{1}{L}&-\frac{1}{L}&-\frac{R}{L} \end{array} \right],
    b_{5} = \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right] \\
    A_{6} &= \left[ \begin{array}{ccc} 0&0&0\\ 0&0&-\frac{1}{C_2} \\ 0&\frac{1}{L}&-\frac{R}{L} \end{array} \right],
    b_{6} = \left[ \begin{array}{c} 0\\ 0\\ 0 \end{array} \right] \\
    A_{7} &= \left[ \begin{array}{ccc} 0&0&0\\ 0&0&0\\ 0&0&-\frac{R}{L} \end{array} \right],
    b_{7} = \left[ \begin{array}{c} 0\\ 0\\ \frac{E}{L} \end{array} \right]
\end{align}

The criteria for the optimization is to compute the trajectory around the average value $\Big[\frac{2}{3}E, \frac{1}{3}E, 1\Big]$ with $E = 30V$.
The maximum cycle period is $T_{p,max} = 0.4ms$ and minimum tweel time of $t_{min} = 0.022ms$. 

The simulation parameters are $C_1 = 40 \mu F$, $C_2 = 40 \mu F$, $L = 10mH$ and $R = 10 \Omega$ with the weight matrix $Q = diag\big[10,5,20000\big]$.

Using the optimal switching sequence
\begin{equation}
    \Omega^\infty = \Big\{ 0, 1, 3, 7, 2, 0, 4, 7, 4 \Big\} \\
\end{equation}

The optimal switching instant is

\begin{equation}
    \tau^\infty = \Big\{ 0, 0.088, 0.110, 0.132, 0.154, 0.176, 0.198, 0.220, 0.242, 0.264 \Big\}
\end{equation}


\noindent
with the initial condition $x(0) = \Big[21.71, 3.29, 1.08\Big]^T$.
