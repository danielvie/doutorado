\section{Statement of the problem}

considerando a planta de um sistema chaveado afim descrito por:

\begin{equation}
	\dot{x}(t) = A_\sigma(t) x(t) + B_\sigma(t)
\end{equation}

submetido a restriçőes de tempo de chaveamento. 

A açăo de controle aplicada no sistema é na forma de perturbacoes no tempo de chaveamento de cada modo de operaçăo, 
e, para podermos controlar o sistema com controle preditivo, precisamos encontrar uma equacao linear para trabalhar a relaçăo
entre os instantes de chaveamento e o estado resultante do sistema ao final de um ciclo.


\textbf{PROBLEMA DE CONTROLE ÓTIMO}

funcao custo

\begin{equation}
	J = \sum_{i=1}^N \hat{e}_{k+i|k} Q \hat{e}^T_{k+i|k} + \rho \hat{\delta t}^2_{k+i|k}
\end{equation}

em que $Q = Q^T > 0$, $\rho > 0$

\textbf{EQUACAO DE PRECIÇĂO}

\begin{equation}
	\bm{\hat{x}} = G \bm{\hat{\delta t}} + \bm{f}
\end{equation}

\begin{equation}
	\boldmath{\hat{x}} = \left[ \begin{array}{c}
		\hat{x}(t_{r_{k + 1}} + \hat{\delta t}_{k + 1 | k}) \\
		\hat{x}(t_{r_{k + 2}} + \hat{\delta t}_{k + 1 | k} + \hat{\delta t}_{k + 2 | k} ) \\
		\vdots \\
		\hat{x}(t_{r_{k + N}} + \hat{\delta t}_{k + 1 | k} + \ldots + \hat{\delta t}_{k + N | k} ) \\
	\end{array} \right]
\end{equation}

\begin{equation}
	\boldmath{\hat{\delta t}} = \left[ \begin{array}{c}
		\hat{\delta t}_{k + 1 | k} \\
		\hat{\delta t}_{k + 2 | k} \\
		\vdots \\
		\hat{\delta t}_{k + N | k}
	\end{array} \right]
\end{equation}

\begin{equation}
	\boldmath{f} = \left[ \begin{array}{c}
		f_{k + 1 | k} \\
		f_{k + 2 | k} \\
		\vdots \\
		f_{k + N | k}
	\end{array} \right]
\end{equation}

\begin{equation}
	G = \left[ \begin{array}{ccccc}
		G_{1,1} & 0 & 0 & \ldots & 0 \\
		G_{2,1} & G_{2,2} & 0 & \ldots & 0 \\
		\vdots & \vdots & \vdots & \vdots & \vdots \\
		G_{N,1} & G_{N,2} & G_{N,3} & \ldots & G_{N,N} \\
	\end{array} \right]
\end{equation}


Now, let $\boldmath{x^r}$ be the vector of target values defined as 

\begin{equation}
	\bm{x^r} = \left[ \begin{array}{ccccc}
		x_r(t_{r_{k+1}}) \\
		x_r(t_{r_{k+2}}) \\
		\vdots  \\
		x_r(t_{r_{k+N}}) \\
	\end{array} \right]
\end{equation}

In view of \red{completar texto aqui}, the cost function is rewritten as 

\begin{equation}
	J = (\bm{\hat{x}} - \bm{x_r})^T \bm{Q} (\bm{\hat{x}} - \bm{x_r}) + \rho \bm{\hat{\delta}t}^T \bm{\hat{\delta}t}
\end{equation}

where $\bm{Q} = \bm{Q}^T > 0$ is an $nN x nN$ block-diagonal matrix defined as $\bm{Q} = diag(Q,Q,...,Q)$.
