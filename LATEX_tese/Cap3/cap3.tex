Cost Equation

\noindent
\begin{equation}
	J = \Vert x(N) \Vert^2_{P_f} \\
	+ \sum^{N - 1}_{k=0} \Big\{\Vert x[k]\Vert ^2_Q + \Vert u[k]\Vert ^2_R\Big\}
\end{equation}

\noindent
\begin{equation}
	\mathbf{u} = \left[
		\begin{matrix}
			u(0) \\
			u(1) \\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right] \in \mathbb{R}^{pN}, 
	\mathbf{x} = \left[
		\begin{matrix}
			x(0) \\
			x(1) \\
			\vdots \\
			x(N)
		\end{matrix}
	\right] \in \mathbb{R}^{nN}
\end{equation}

\noindent
com isso, a somatória

\noindent
\begin{equation}
	\sum^{N - 1}_{k=0} \Vert u(k)\Vert ^2_R = u^T(0) R u(0) + u^T(1) R u(1) + \dotsi + u^T(N-1) R u(N-1)
\end{equation}

\noindent
fica na forma

\noindent
\begin{equation}
	\underbrace{
			\left[
			\begin{matrix}
				u^T(0) & u^T(1) & \dotsi & u^T(N-1)
			\end{matrix}
			\right]
	}_{\mathbf{u}^T}
	\underbrace{
	\left[
		\begin{matrix}
			R & 0 & \dotsi & 0 \\
			0 & R & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & R
		\end{matrix}
	\right]
	}_{\mathbf{R}}
	\underbrace{
		\left[
			\begin{matrix}
				u(0) \\
				u(1) \\
				\dotsi \\
				u(N-1)
			\end{matrix}
		\right]
	}_{\mathbf{u}}
\end{equation}

\noindent
pode-se escrever

\noindent
\begin{equation}
	\sum^{N - 1}_{k=0} \Vert u(k)\Vert ^2_R = \Vert \mathbf{u} \Vert^2_{\mathbf{R}}
\end{equation}

\noindent
de forma similar, pode-se escrever

\noindent
\begin{equation}
	\Vert x(N) \Vert^2_{P_f} + \sum^{N - 1}_{k=0} \Vert x(k)\Vert ^2_Q = x^T(0) Q x(0) + x^T(1) Q x(1) + \dotsi + x^T(N-1) Q x(N-1) + x^T(N) P_f x(N)
\end{equation}

\noindent
\begin{equation}
	\underbrace{
			\left[
			\begin{matrix}
				u^T(1) & u^T(2) & \dotsi & u^T(N)
			\end{matrix}
			\right]
	}_{\mathbf{x}^T}
	\underbrace{
	\left[
		\begin{matrix}
			Q & 0 & \dotsi & 0 & 0 \\
			0 & Q & \dotsi & 0 & 0 \\
			\vdots & \vdots & \ddots & \vdots & \vdots \\
			0 & 0 & \dotsi & Q & 0 \\
			0 & 0 & \dotsi & 0 & P_f \\
		\end{matrix}
	\right]
	}_{\mathbf{Q}}
	\underbrace{
		\left[
			\begin{matrix}
				x(1) \\
				x(2) \\
				\dotsi \\
				x(N)
			\end{matrix}
		\right]
	}_{\mathbf{x}}
\end{equation}

\noindent
de forma abreviada

\noindent
\begin{equation}
	\mathbf{Q} = 
	\left[
		\begin{matrix}
			diag(Q) & 0 \\
			0 & P_f \\
		\end{matrix}
	\right]
\end{equation}

\noindent
pode-se escrever

\noindent
\begin{equation}
	\Vert x(N) \Vert^2_{P_f} + \sum^{N - 1}_{k=0} \Vert x(k)\Vert ^2_Q = \Vert x(0) \Vert^2_Q + \Vert \mathbf{x} \Vert^2_{\mathbf{Q}}
\end{equation}

Em resumo, o custo pode ser escrito com 

\noindent
\begin{equation}
	J(\mathbf{x}, \mathbf{u}) = \Vert x(0) \Vert^2_Q + \Vert {\mathbf{x}} \Vert^2_{\mathbf{Q}} + \Vert \mathbf{u} \Vert^2_{\mathbf{R}}			
\end{equation}

\noindent
\begin{equation}
   u = \left[
    \begin{matrix}
        u(0) \\
        u(1) \\
        \dotsi \\
        u(N-1)
    \end{matrix}
   \right], 
   x = 
   \left[
    \begin{matrix}
        x(1) \\
        x(2) \\
        \dotsi \\
        x(N)
    \end{matrix}
   \right]
\end{equation}

\noindent
com base na equacao de estado

\noindent
\begin{equation}
    x(k+1) = Ax(k) + Bu(k)
\end{equation}

\noindent
e na condicao inicial x(0)

Nesse ponto, vale lembrar a solucao da equacao de estado:

\noindent
\begin{equation}
    x(k) = A^k x(0) + \sum^{k-1}_{i=0}{A^{k-1-i}Bu(i)}
\end{equation}

que pode ser escrita para k = 1,2,...,N como



\noindent
\begin{align}
        x(1) &= Ax(0) + Bu(0) \\
        x(2) &= A^2x(0) + ABu(0) + Bu(1) \\
        \vdots \\
        x(N) &= A^Nx(0) + A^{N-1}Bu(0) + \dotsi + ABu(N-2) + Bu(N - 1)
\end{align}

ou ainda

\noindent
\begin{equation}
    \left[
        \begin{matrix}
            x(1) \\
            x(2) \\
            \vdots \\
            x(N)
        \end{matrix}
    \right] = 
    \left[
        \begin{matrix}
            A \\
            A^2 \\
            \vdots \\
            A^N
        \end{matrix}
    \right] x(0) + 
    \left[
        \begin{matrix}
           B & 0 & \dotsi & 0 \\ 
           AB & B & \dotsi & 0 \\ 
           \vdots & \vdots & \ddots & \vdots \\ 
           A^{N-1}B & A^{N-2}B & \dotsi & B
        \end{matrix}
    \right]
    \left[
        \begin{matrix}
            u(0) \\
            u(1) \\
            \vdots \\
            u(N-1)
        \end{matrix}
    \right]
\end{equation}

De forma mais concisa, vamos escrever

\noindent
\begin{equation}
    \mathbf{x} = \mathbf{A} x(0) + \mathbf{B} \mathbf{u}
\end{equation}

\noindent
com

\noindent
\begin{equation}
    \mathbf{A} = \left[
        \begin{matrix}
            A \\
            A^2 \\
            \vdots \\
            A^N
        \end{matrix}
    \right],
    \mathbf{B} = \left[
        \begin{matrix}
           B & 0 & \dotsi & 0 \\
           AB & B & \dotsi & 0 \\
           \vdots & \vdots & \ddots & \vdots \\
           A^{N-1}B & A^{N-2}B & \dotsi & B
        \end{matrix}
    \right]
\end{equation}

Tendo em vista o vínculo

\noindent
\begin{equation}
    \mathbf{x} = \mathbf{A}x(0) + \mathbf{B} \mathbf{u}
\end{equation}

\noindent
pode-se reescrever a funcao de custo como 

\noindent
\begin{equation}
    \begin{gathered}
        J(\mathbf{x}, \mathbf{u}) = \Vert x(0) - \bar{x} \Vert^2_Q + \Vert x - col_N(\bar{x}) \Vert^2_{\mathbf{Q}} + \Vert \mathbf{u} - col_N(\bar{u}) \Vert^2_{\mathbf{R}} \\
        = \Vert x(0) - \bar{x} \Vert^2_Q \\
        + [\mathbf{A}x(0) + \mathbf{B}\mathbf{u} - col_N(\bar{x})] \\
        + [\mathbf{u} - col_N(\bar{u})]^T \mathbf{R}[\mathbf{u} - col_N(\bar{u})] \\
        J(\mathbf{x}, \mathbf{u}) = \Vert x(0) - \bar{x} \vert^2_Q \\
        + [\mathbf{A} x(0) - col_N(\bar{x}) + \mathbf{B} \mathbf{u}]^T \mathbf{Q} [\mathbf{A} x(0) - col_N(\bar{x}) \mathbf{B} \mathbf{u}] \\
        + [\mathbf{u} - col_N(\bar{u})]^T\mathbf{R}[\mathbf{u} - col_N(\bar{u})] \\
        = \Vert x(0) - \bar{x} \Vert^2_Q + \Vert \mathbf{A} x(0) - col_N(\bar{x}) \Vert^2_{\mathbf{Q}} \\
        + [\mathbf{A} x(0) - col_N(\mathbf{x})]^T \mathbf{Q} \mathbf{B} \mathbf{u} + \mathbf{u}^T\mathbf{B}^T\mathbf{Q}[\mathbf{A} x(0) - col_N(\bar{x})] + \mathbf{u}^T\mathbf{B}^T\mathbf{Q}\mathbf{B}\mathbf{u} \\
        + \mathbf{u}^T \mathbf{R} \mathbf{u} - \mathbf{u}^T\mathbf{R}[col_N(\bar{u})] - [col_N(\bar{u})]^T \mathbf{R} \mathbf{u} + \Vert col_N (\bar{u}) \Vert^2_{\mathbf{R}} \\
        J(\mathbf{x}, \mathbf{u}) = \blue{\Vert x(0) - \bar{x} \Vert^2_Q + \Vert\mathbf{A}x(0) - col_N(\bar{x}) \Vert^2_{\mathbf{Q}} + \Vert col_N(\bar{u}) \Vert^2_{\mathbf{R}}} \\
        + \darkgreen{[\mathbf{A} - col_N(\bar{x})]^T \mathbf{Q} \mathbf{B} \mathbf{u} + \mathbf{u}^T \mathbf{B}^T \mathbf{Q} [\mathbf{A} x(0) - col_N(\bar{x})]} \\
        - \mathbf{u}^T\mathbf{R} [ col_N(\bar{u})] - [col_N(\bar{u})]^T \mathbf{R} \mathbf{u} + \mathbf{u}^T (\mathbf{R} + \mathbf{B}^T \mathbf{Q} \mathbf{B}) \mathbf{u}
    \end{gathered}
\end{equation}

Observaçoes:

- O \blue{termo constante} será denotado por \blue{$c_0$}

- Lembrando que a matriz $\mathbf{Q}$ é simétrica, tem-se

\begin{equation}
    \darkgreen{[\mathbf{A} x(0) - col_N(\bar{x})]^T\mathbf{Q}\mathbf{B}\mathbf{u} = \Big\{ \mathbf{u}^T \mathbf{B}^T \mathbf{Q} [\mathbf{A} x(0) - col_N(\bar{x})] \Big\}^T}
\end{equation}

Como esses termos sao escalares, conclui-se que sao iguais entre si. Logo:

\begin{equation}
    \begin{gathered}
        \darkgreen{\mathbf{A} x(0) - col_N(\bar{x})^T \mathbf{Q} \mathbf{B} \mathbf{u} + \mathbf{u}^T\mathbf{B}^T\mathbf{Q}[\mathbf{A}x(0) - col_N(\bar{x})]} \\
        \darkgreen{= 2 [\mathbf{A} x(0) - col_N(\bar{x})]^T\mathbf{Q}\mathbf{B}\mathbf{u}}
    \end{gathered}
\end{equation}

De forma similar:

\begin{equation}
    -\mathbf{u}^T\mathbf{R}[col_N(\bar{u})] - [col_N(\bar{u})]^T\mathbf{R}\mathbf{u} = -2[col_N(\bar{u})]^T\mathbf{R}\mathbf{u}
\end{equation}


Finalmente, pode-se escrever

\begin{equation}
    \begin{gathered}
        J(\mathbf{u}) = \mathbf{u}^T(\mathbf{B}^T\mathbf{Q}\mathbf{B} + \mathbf{R})\mathbf{u} \\
        + 2 \Big\{[\mathbf{A}x(0) - col_N(\bar{x})]^T \mathbf{Q} \mathbf{B} - [col_N(\bar{u})]^T \mathbf{R} \Big\} \mathbf{u} + c_0
    \end{gathered}
\end{equation}

\noindent
com

\begin{equation}
    c_0 = \Vert x(0) - \bar{x} \Vert^2_Q + \Vert \mathbf{A} x(0) - col_N (\bar{x}) \Vert^2_{\mathbf{R}} + \Vert col_N (\bar{u}) \Vert^2_{\mathbf{R}}
\end{equation}

Note-se que passaremos a usar a notaçao $J(\mathbf{u})$ em lugar de $J(\mathbf{x}, \mathbf{u})$, uma vez que $\mathbf{x}$ foi eliminada da expressao do custo.

Para terminar a reformulacao do problema de otimizacao em termos de $\mathbf{u}$, deve-se considerar o vinculo entre \textbf{x} e \textbf{u} nas \textbf{restricoes}

Considerando novamente o vinculo

\begin{equation}
    \mathbf{x} = \mathbf{A} x(0) + \mathbf{B} \mathbf{u}
\end{equation}

\noindent
a restricao

\begin{equation}
    \mathbf{S_x} \mathbf{x} \le \mathbf{b_x}
\end{equation}

\noindent
pode ser reformulada como

\begin{equation}
    \mathbf{S_x} \mathbf{B} \mathbf{u} \le \mathbf{b_x} - \mathbf{S_x} \mathbf{A} x(0)
\end{equation}

Em conjunto com $\mathbf{S_u} \mathbf{u} \le \mathbf{b_u}$, pode-se entao expressar as restricoes na forma

\begin{equation}
    \left[
        \begin{matrix}
            \mathbf{S_x} \mathbf{B} \\
            \mathbf{S_u}
        \end{matrix} \mathbf{u} 
    \right] \le
    \left[
        \begin{matrix}
            \mathbf{b_x} - \mathbf{S_x} \mathbf{A} x(0) \\
            \mathbf{b_u}
        \end{matrix}
    \right]
\end{equation}
