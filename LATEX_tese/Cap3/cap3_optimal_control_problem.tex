\subsection{{Prediction equation}}

\purple{INICIO referencia nova}

Cost Equation

\noindent
\begin{equation}
	J = \Vert x(N) \Vert^2_{P_f} \\
	+ \sum^{N - 1}_{k=0}\left[\Vert x(k)\Vert ^2_Q + \Vert u(k)\Vert ^2_R\right]
\end{equation}

\noindent
\begin{equation}
	\mathbf{u} = \left[
		\begin{matrix}
			u(0) \\
			u(1) \\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right] \in \mathbb{R}^{pN}, 
	\mathbf{x} = \left[
		\begin{matrix}
			x(0) \\
			x(1) \\
			\vdots \\
			x(N)
		\end{matrix}
	\right] \in \mathbb{R}^{nN}
\end{equation}

\noindent
com isso, a somatória

\noindent
\begin{equation}
	\sum^{N - 1}_{k=0} \Vert u(k)\Vert ^2_R = u^T(0) R u(0) + u^T(1) R u(1) + \dotsi + u^T(N-1) R u(N-1)
\end{equation}

\noindent
fica na forma

\noindent
\begin{equation}
	\underbrace{
			\left[
			\begin{matrix}
				u^T(0) & u^T(1) & \dotsi & u^T(N-1)
			\end{matrix}
			\right]
	}_{\mathbf{u}^T}
	\underbrace{
	\left[
		\begin{matrix}
			R & 0 & \dotsi & 0 \\
			0 & R & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & R
		\end{matrix}
	\right]
	}_{\mathbf{R}}
	\underbrace{
		\left[
			\begin{matrix}
				u(0) \\
				u(1) \\
				\dotsi \\
				u(N-1)
			\end{matrix}
		\right]
	}_{\mathbf{u}}
\end{equation}

\noindent
pode-se escrever

\noindent
\begin{equation}
	\sum^{N - 1}_{k=0} \Vert u(k)\Vert ^2_R = \Vert \mathbf{u} \Vert^2_{\mathbf{R}}
\end{equation}

\noindent
de forma similar, pode-se escrever

\noindent
\begin{equation}
	\Vert x(N) \Vert^2_{P_f} + \sum^{N - 1}_{k=0} \Vert x(k)\Vert ^2_Q = x^T(0) Q x(0) + x^T(1) Q x(1) + \dotsi + x^T(N-1) Q x(N-1) + x^T(N) P_f x(N)
\end{equation}

\noindent
\begin{equation}
	\underbrace{
			\left[
			\begin{matrix}
				u^T(1) & u^T(2) & \dotsi & u^T(N)
			\end{matrix}
			\right]
	}_{\mathbf{x}^T}
	\underbrace{
	\left[
		\begin{matrix}
			Q & 0 & \dotsi & 0 & 0 \\
			0 & Q & \dotsi & 0 & 0 \\
			\vdots & \vdots & \ddots & \vdots & \vdots \\
			0 & 0 & \dotsi & Q & 0 \\
			0 & 0 & \dotsi & 0 & P_f \\
		\end{matrix}
	\right]
	}_{\mathbf{Q}}
	\underbrace{
		\left[
			\begin{matrix}
				x(1) \\
				x(2) \\
				\dotsi \\
				x(N)
			\end{matrix}
		\right]
	}_{\mathbf{x}}
\end{equation}

\noindent
de forma abreviada

\noindent
\begin{equation}
	\mathbf{Q} = 
	\left[
		\begin{matrix}
			diag(Q) & 0 \\
			0 & P_f \\
		\end{matrix}
	\right]
\end{equation}

\noindent
pode-se escrever

\noindent
\begin{equation}
	\Vert x(N) \Vert^2_{P_f} + \sum^{N - 1}_{k=0} \Vert x(k)\Vert ^2_Q = \Vert x(0) \Vert^2_Q + \Vert \mathbf{x} \Vert^2_{\mathbf{Q}}
\end{equation}

\noindent
em resumo, o custo pode ser escrito com 

\noindent
\begin{equation}
	J(\mathbf{x}, \mathbf{u}) = \Vert x(0) \Vert^2_Q + \Vert {\mathbf{x}} \Vert^2_{\mathbf{Q}} + \Vert \mathbf{u} \Vert^2_{\mathbf{R}}			
\end{equation}
