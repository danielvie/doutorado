\subsection{{Prediction equation}}

\begin{equation}
	\bm{\hat{x}} = G \bm{\hat{\delta t}} + \bm{f}
\end{equation}

\begin{equation}
	\boldmath{\hat{x}} = \left[ \begin{array}{c}
		\hat{x}(t_{r_{k + 1}} + \hat{\delta t}_{k + 1 | k}) \\
		\hat{x}(t_{r_{k + 2}} + \hat{\delta t}_{k + 1 | k} + \hat{\delta t}_{k + 2 | k} ) \\
		\vdots \\
		\hat{x}(t_{r_{k + N}} + \hat{\delta t}_{k + 1 | k} + \ldots + \hat{\delta t}_{k + N | k} ) \\
	\end{array} \right]
\end{equation}

\begin{equation}
	\boldmath{\hat{\delta t}} = \left[ \begin{array}{c}
		\hat{\delta t}_{k + 1 | k} \\
		\hat{\delta t}_{k + 2 | k} \\
		\vdots \\
		\hat{\delta t}_{k + N | k}
	\end{array} \right]
\end{equation}

\begin{equation}
	\boldmath{f} = \left[ \begin{array}{c}
		f_{k + 1 | k} \\
		f_{k + 2 | k} \\
		\vdots \\
		f_{k + N | k}
	\end{array} \right]
\end{equation}

\begin{equation}
	G = \left[ \begin{array}{ccccc}
		G_{1,1} & 0 & 0 & \ldots & 0 \\
		G_{2,1} & G_{2,2} & 0 & \ldots & 0 \\
		\vdots & \vdots & \vdots & \vdots & \vdots \\
		G_{N,1} & G_{N,2} & G_{N,3} & \ldots & G_{N,N} \\
	\end{array} \right]
\end{equation}
