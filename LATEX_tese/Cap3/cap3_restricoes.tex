
\section{Restrictions}

The restrictions are introduced aiming to drive the states of the system to the origin while solving

\begin{equation}
	\min_{\mathbf{x} \in \mathbb{R}^{nN}, \mathbf{u} \in \mathbb{R}^{pN}}{\mathbf{J(x,u)}}
\end{equation}

\noindent
respecting constraints applied on

\noindent
\begin{align}
	\mathbf{u} &= \left[
		\begin{matrix}
			u(0) \\
			u(1) \\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right]
	\in \mathcal{U} \times  \mathcal{U} \times \dotsi \times \mathcal{U} = \mathcal{U}^N \\
	\mathbf{x} &= \left[
		\begin{matrix}
			x(1) \\
			x(2) \\
			\vdots \\
			x(N)
		\end{matrix}
	\right]
	\in \mathcal{X} \times  \mathcal{X} \times \dotsi \times \mathcal{X}_f = \mathcal{X}^{N-1} \times \mathcal{X}_f
\end{align}

\noindent

The constraint on $\mathbf{u}$ can be  written as
\begin{equation}
	\left[
		\begin{matrix}
			S_u & 0 & \dotsi & 0 \\
			0 & S_u & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & S_u
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			u(0) \\
			u(1) \\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right]
	\leq
	\left[
		\begin{matrix}
			b_u \\
			b_u \\
			\vdots \\
			b_u
		\end{matrix}
	\right]
\end{equation}

\noindent
or, in a shorter form

\begin{equation}
	\mathbf{S_u u} \leq \mathbf{b_u}
\end{equation}

\noindent
with

\begin{equation}
	\mathbf{S_u} = diag_N(S_u), \mathbf{b_u} = col_N(b_u)
\end{equation}

Similarly, the constraints on $\mathbf{x}$ can be written as

\begin{equation}
	\left[
		\begin{matrix}
			S_x & 0 & \dotsi & 0 \\
			0 & S_x & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			0 & 0 & \dotsi & S_x
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			x(1) \\
			x(2) \\
			\vdots \\
			x(N)
		\end{matrix}
	\right]
	\leq
	\left[
		\begin{matrix}
			b_x \\
			b_x \\
			\vdots \\
			b_x
		\end{matrix}
	\right]
\end{equation}

\noindent
or, in short

\begin{equation}
	\mathbf{S_x x} \leq \mathbf{b_x}
\end{equation}

\noindent
with

\begin{equation}
	\mathbf{S_x} = 
	\left[
		\begin{matrix}
			diag_{N-1}(S_x) & 0 \\
			0 & S_f
		\end{matrix}
	\right], \mathbf{b_x} = 
	\left[
		\begin{matrix}
			col_{N-1}(b_x) \\
			b_f
		\end{matrix}
	\right]
\end{equation}
