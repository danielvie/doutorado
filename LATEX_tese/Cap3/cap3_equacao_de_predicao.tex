\section{Prediction Equation}

Let's express the relationship between the vectors

\begin{equation}
	\mathbf{\hat{u}} = \left[
		\begin{matrix}
			\hat{u}(0) \\ 
			\hat{u}(1) \\ 
			\vdots \\ 
			\hat{u}(N-1)
		\end{matrix}
	\right],
	\mathbf{\hat{x}} = \left[
		\begin{matrix}
			\hat{x}(1) \\ 
			\hat{x}(2) \\ 
			\vdots \\ 
			\hat{x}(N)
		\end{matrix}
	\right]
\end{equation}

\purple{
	\begin{equation}
		\mathbf{\hat{u}} = \left[
			\begin{matrix}
				\hat{u}(k | k) \\ 
				\hat{u}(k + 1 | k) \\ 
				\vdots \\ 
				\hat{u}(k + N-1 | k)
			\end{matrix}
		\right],
		\mathbf{\hat{x}} = \left[
			\begin{matrix}
				\hat{x}(k + 1 | k) \\ 
				\hat{x}(k + 2 | k) \\ 
				\vdots \\ 
				\hat{x}(k + N | k)
			\end{matrix}
		\right]
		\nonumber
	\end{equation}
}

\noindent
based on the state equation

\begin{equation}
	\hat{x}(k+1) = A\hat{x}(k) + B \hat{u}(k)
\end{equation}

\noindent
and initial condition $x(0)$.

\purple{
	\noindent
	and initial condition $x(k)$.
}

The solution for the state equation is:

\begin{equation}
	\hat{x}(k) = A^kx(k) + \sum_{i=0}^{k-1} A^{k-1-i} B \hat{u}(i)
\end{equation}

\noindent
and it can be written for $k = 1,2,...,N$ as

\noindent
\begin{align}
	\hat{x}(1) &= A x(0) +  B \hat{u}(0) \\
	\hat{x}(2) &= A^2 x(0) + A B \hat{u}(0) +  B \hat{u}(1) \\
	&\vdots \\
	\hat{x}(N) &= A^N x(0) A^{N - 1} B \hat{u}(0) + \dotsi + A B \hat{u}(N - 2) +  B \hat{u}(N - 1)
\end{align}

\noindent
or

\begin{equation}
	\left[
		\begin{matrix}
			\hat{x}(1) \\
			\hat{x}(2) \\
			\vdots \\
			\hat{x}(N) \\
		\end{matrix}
	\right] = 
	\left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	x(0) + 
	\left[
		\begin{matrix}
			B & 0 & \dotsi & 0 \\
			AB & B & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \dotsi & B
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			\hat{u}(0) \\
			\hat{u}(1)\\
			\vdots \\
			\hat{u}(N-1)
		\end{matrix}
	\right]
\end{equation}

This can be written in a shorter form as

\begin{equation}
	\mathbf{\hat{x}} = \mathbf{A}x(0) + \mathbf{B \hat{u}}
\end{equation}

\noindent
with 

\begin{equation}
	\mathbf{A} = \left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	, 
	\mathbf{B} = \left[
		\begin{matrix}
			B & 0 & \dotsi & 0 \\
			AB & B & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \dotsi & B
		\end{matrix}
	\right]
\end{equation}

The relation

\begin{equation}
	\mathbf{\hat{x}} = \mathbf{A}x(0) + \mathbf{B \hat{u}}
\end{equation}

can be rewritten as

\noindent
\begin{equation}
	J = \norma{x(0)}{P}
		+ \green{\overbrace{\norma{\mathbf{\hat{x}}}{\mathbf{Q}}}^{\square}} 
		+ \norma{\mathbf{\hat{u}}}{\mathbf{R}}
\end{equation}

\noindent
\begin{align}
	\green{\square} &= \green{\norma{\mathbf{\hat{x}}}{\mathbf{Q}}} \\
	&= \green{
		\Big[
			\mathbf{A}x(0) + \mathbf{B \hat{u}}
		\Big]^T \mathbf{Q}
		\Big[
			\mathbf{A}x(0) + \mathbf{B \hat{u}}
		\Big]
	} \\
	&= \green{
		\Big[
			\mathbf{A}x(0) + \mathbf{B \hat{u}}
		\Big]^T \mathbf{Q}
		\Big[
			\mathbf{A}x(0) + \mathbf{B \hat{u}}
		\Big]
	} \\
	\begin{split}
		&= \green{
			\norma{\mathbf{A}x(0)}{\mathbf{Q}}
		} \green{ + 
			\Big[
				\mathbf{A}x(0)
			\Big]^T \mathbf{Q} \mathbf{B \hat{u}}
		} \\
		&\qquad \green{ + 
			\mathbf{\hat{u}}^T \mathbf{B}^T \mathbf{Q}
			\mathbf{A}x(0)
		} \green{ + 
			\mathbf{\hat{u}}^T \mathbf{B}^T \mathbf{Q} \mathbf{B \hat{u}}
		}
	\end{split}
\end{align}

replacing $\green{\square}$ back in the cost equation and reordering the elements 

\noindent
\begin{align}
	J(\mathbf{\hat{x}}, \mathbf{\hat{u}}) 
	&= \purple{ 
			\norma{x(0)}{P} 
			+ \norma{\mathbf{A}x(0)}{\mathbf{Q}} 
		} \\
	&\quad + \orange{
			\Big[\mathbf{A}x(0)\Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}	
			+ \mathbf{\hat{u}}^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x(0)
		} \\
	&\quad + \blue{
			\mathbf{\hat{u}}^T \mathbf{R} \mathbf{\hat{u}} 
			+ \mathbf{\hat{u}}^T \mathbf{B}^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}
		}
\end{align}

The \purple{constant part} will be denoted by \purple{$c_0$}

Recalling that the matrix $\mathbf{Q}$ is symmetric, we have:

\begin{equation}
	\orange{
		\Big[ \mathbf{A}x(0)  \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}} 
	} = \Big\{ 
	\orange{
		\mathbf{\hat{u}}^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x(0) 
	} \Big\}^T
\end{equation}

As these terms are scalars, it can be concluded that they are equal to each other, therefore:

\noindent
\begin{equation}
	\begin{gathered}
		\orange{
			\Big[ \mathbf{A}x(0)  \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}} + \mathbf{\hat{u}}^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x(0) }
		\\
		=
		\orange{
			2\Big[ \mathbf{A}x(0)  \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}} 
		}
	\end{gathered}
\end{equation}

Finally, it can be written

\noindent
\begin{equation}
	\begin{gathered}
		J(\mathbf{\hat{u}}) = 
			\blue{
				\mathbf{\hat{u}}^T (\mathbf{R} + \mathbf{B}^T \mathbf{Q} \mathbf{B}) \mathbf{\hat{u}}
			}
			+ \orange{
				2 \Big[ \mathbf{A}x(0) \Big]^T \mathbf{Q} \mathbf{B} \mathbf{\hat{u}}
			}
			+ \purple{
				c_0
			}
	\end{gathered}
\end{equation}

with

\noindent
\begin{equation}
	\purple{
		c_0
	} = 
	\purple{
		\norma{x(0)}{P} + \norma{\mathbf{A} x(0)}{\mathbf{Q}}
	}
\end{equation}

Note that we will now use the notation $J(\mathbf{\hat{u}})$ instead of $J(\mathbf{\hat{x}}, \mathbf{\hat{u}})$, since $\mathbf{\hat{x}}$ has been eliminated from 
the cost expression. 

To complete the reformulation of the optimization problem in terms of $\mathbf{\hat{u}}$, the relation between $\mathbf{\hat{x}}$ and $\mathbf{\hat{u}}$ in the 
\textbf{restrictions} must be considered.
