\section{Prediction Equation}

Let's express the relationship between the vectors

\begin{equation}
	\mathbf{u} = \left[
		\begin{matrix}
			u(0) \\ 
			u(1) \\ 
			\vdots \\ 
			u(N-1)
		\end{matrix}
	\right],
	\mathbf{x} = \left[
		\begin{matrix}
			x(1) \\ 
			x(2) \\ 
			\vdots \\ 
			x(N)
		\end{matrix}
	\right]
\end{equation}

\noindent
based on the state equation

\begin{equation}
	x(k+1) = Ax(k) + Bu(k)
\end{equation}

\noindent
and initial condition $x(0)$.

The solution for the state equation is:

\begin{equation}
	x(k) = A^kx(0) + \sum_{i=0}^{k-1} A^{k-1-i} B u(i)
\end{equation}

\noindent
and it can be written for $k = 1,2,...,N$ as

\noindent
\begin{align}
	x(1) &= A x(0) + Bu(0) \\
	x(2) &= A^2 x(0) + ABu(0) + Bu(1) \\
	&\vdots \\
	x(N) &= A^N x(0) A^{N - 1}Bu(0) + \dotsi + ABu(N - 2) + Bu(N - 1)
\end{align}

\noindent
or

\begin{equation}
	\left[
		\begin{matrix}
			x(1) \\
			x(2) \\
			\vdots \\
			x(N) \\
		\end{matrix}
	\right] = 
	\left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	x(0) + 
	\left[
		\begin{matrix}
			B & 0 & \dotsi & 0 \\
			AB & B & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \dotsi & B
		\end{matrix}
	\right]
	\left[
		\begin{matrix}
			u(0) \\
			u(1)\\
			\vdots \\
			u(N-1)
		\end{matrix}
	\right]
\end{equation}

This can be written in a shorter form as

\begin{equation}
	\mathbf{x} = \mathbf{A}x(0) + \mathbf{Bu}
\end{equation}

\noindent
with 

\begin{equation}
	\mathbf{A} = \left[
		\begin{matrix}
			A \\
			A^2 \\
			\vdots \\
			A^N
		\end{matrix}
	\right]
	, 
	\mathbf{B} = \left[
		\begin{matrix}
			B & 0 & \dotsi & 0 \\
			AB & B & \dotsi & 0 \\
			\vdots & \vdots & \ddots & \vdots \\
			A^{N-1}B & A^{N-2}B & \dotsi & B
		\end{matrix}
	\right]
\end{equation}

The relation

\begin{equation}
	\mathbf{x} = \mathbf{A}x(0) + \mathbf{Bu}
\end{equation}

can be rewritten as

\begin{equation}
		J(\mathbf{x}, \mathbf{u}) = \norma{x(0) - \bar{x}}{P} 
		+ \green{\overbrace{\norma{\mathbf{x} - col_N(\bar{x})}{\mathbf{Q}}}^\square} 
		+ \blue{\overbrace{\norma{\mathbf{u} - col_N(\bar{u})}{\mathbf{R}}}^\bigcirc}
\end{equation}

\purple{
	===============================

	ajustar nomenclatura e organizar termos

	===============================
}

\noindent
\begin{equation}
	J = \Vert x(0) \Vert^2_{P} 
		+ \green{\overbrace{\norma{\mathbf{x}}{\mathbf{Q}}}^{\square}} 
		+ \norma{\mathbf{u}}{\mathbf{R}}
\end{equation}

\noindent
\begin{align}
	\green{\square} &= \green{\norma{\mathbf{x}}{\mathbf{Q}}} \\
	&= \green{
		\Big[
			\mathbf{A}x(0) + \mathbf{Bu})
		\Big]^T \mathbf{Q}
		\Big[
			\mathbf{A}x(0) + \mathbf{Bu})
		\Big]
	} \\
	&= \green{
		\Big[
			\mathbf{A}x(0) + \mathbf{Bu}
		\Big]^T \mathbf{Q}
		\Big[
			\mathbf{A}x(0) + \mathbf{Bu}
		\Big]
	} \\
	\begin{split}
		&= \green{
			\norma{\mathbf{A}x(0)}{\mathbf{Q}}
		} \green{ + 
			\Big[
				\mathbf{A}x(0)
			\Big]^T \mathbf{Q} \mathbf{Bu}
		} \\
		&\qquad \green{ + 
			\mathbf{u}^T \mathbf{B}^T \mathbf{Q}
			\mathbf{A}x(0)
		} \green{ + 
			\mathbf{u}^T \mathbf{B}^T \mathbf{Q} \mathbf{Bu}
		}
	\end{split}
\end{align}

replacing $\green{\square}$ back in the cost equation and reordering the elements 

\noindent
\begin{align}
	J(\mathbf{x}, \mathbf{u}) 
	&= \purple{ 
			\norma{x(0)}{P} 
			+ \norma{\mathbf{A}x(0)}{\mathbf{Q}} 
		} \\
	&\quad + \orange{
			\Big[\mathbf{A}x(0)\Big]^T \mathbf{Q} \mathbf{B} \mathbf{u}	
			+ \mathbf{u}^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x(0)
		} \\
	&\quad + \blue{
			\mathbf{u}^T \mathbf{R} \mathbf{u} 
			+ \mathbf{u}^T \mathbf{B}^T \mathbf{Q} \mathbf{B} \mathbf{u}
		}
\end{align}

The \purple{constant part} will be denoted by \purple{$c_0$}

Recalling that the matrix $\mathbf{Q}$ is symmetric, we have:

\begin{equation}
	\orange{
		\Big[ \mathbf{A}x(0)  \Big]^T \mathbf{Q} \mathbf{B} \mathbf{u} 
	} = \Big\{ 
	\orange{
		\mathbf{u}^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x(0) 
	} \Big\}^T
\end{equation}

As these terms are scalars, it can be concluded that they are equal to each other, therefore:

\noindent
\begin{equation}
	\begin{gathered}
		\orange{
			\Big[ \mathbf{A}x(0)  \Big]^T \mathbf{Q} \mathbf{B} \mathbf{u} + \mathbf{u}^T \mathbf{B}^T \mathbf{Q} \mathbf{A} x(0) }
		\\
		=
		\orange{
			2\Big[ \mathbf{A}x(0)  \Big]^T \mathbf{Q} \mathbf{B} \mathbf{u} 
		}
	\end{gathered}
\end{equation}

Finally, it can be written

\noindent
\begin{equation}
	\begin{gathered}
		J(\mathbf{u}) = 
			\blue{
				\mathbf{u}^T (\mathbf{R} \mathbf{B}^T \mathbf{Q} \mathbf{B}) \mathbf{u}
			}
			+ \orange{
				2 \Big[ \mathbf{A}x(0) \Big]^T \mathbf{Q} \mathbf{B} \mathbf{u}
			}
			+ \purple{
				c_0
			}
	\end{gathered}
\end{equation}

with

\noindent
\begin{equation}
	\purple{
		c_0
	} = 
	\purple{
		\norma{x(0)}{P} + \norma{\mathbf{A} x(0)}{\mathbf{Q}}
	}
\end{equation}

Note that we will now use the notation $J(\mathbf{u})$ instead of $J(\mathbf{x}, \mathbf{u})$, since $\mathbf{x}$ has been eliminated from 
the cost expression. 

To complete the reformulation of the optimization problem in terms of $\mathbf{u}$, the relation between $\mathbf{x}$ and $\mathbf{u}$ in the 
\textbf{restrictions} must be considered.
