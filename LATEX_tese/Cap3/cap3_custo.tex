
Consider a system with dynamics described by

\begin{equation}
    \hat{x}(k) = A x(k) + B \hat{u}(k)
\end{equation}

The objective is to drive the state $\hat{x}(K)$ to the origin from a given initial condition while respecting
state and control constraints in the form

\begin{equation}
    \hat{x}(k) \in \mathcal{X}
\end{equation}
    
\begin{equation}
    \hat{u}(k) \in \mathcal{U}
\end{equation}


\begin{equation}
    \mathcal{U} = \left\{
        \hat{u} \in \mathbb{R}^p : S_u \hat{u} \leq b_u	
    \right\}
\end{equation}

\noindent
and

\begin{equation}
    \mathcal{X} = \left\{
        \hat{x} \in \mathbb{R}^n : S_x \hat{x} \leq b_x
    \right\}
\end{equation}


\section{Cost Function}

The solution, if it exists, will not be unique. Therefore, it is necessary to adopt a criterion that allows for choosing the best sequence of control 
actions to perform the task.


The control problem consists of bringing the states to the origin, so we will employ a cost in the form

\begin{equation}
    J = \blue{\Vert \hat{x}(N) \Vert^2_{P_f}} + \orange{\sum^{N-1}_{k=0} \Big[ \Vert \hat{x}(k) \Vert^2_Q +  \Vert \hat{u}(k) \Vert^2_R \Big]}
\end{equation}

The expression for $J$ consists of a \blue{terminal cost} term and a trajectory \orange{cost term}. 

With $P_f$, $Q$, $R$ being symmetric and positive-definite, with $P_f$ chosen so that the terminal cost term $\norma{\hat{x}(k + N|k) - \bar{x}}{P_f}$ corresponds 
to the value function of the infinite horizon Discrete-Time Linear Quadratic Regulator (DLQR).

The cost function can be rewritten using the notation:

\begin{equation}
    \mathbf{\hat{u}} = 
    \left[
        \begin{matrix}
            \hat{u}(0) \\		
            \hat{u}(1) \\		
            \vdots \\
            \hat{u}(N-1)
        \end{matrix}
    \right]
    \in \mathbb{R}^{pN},
    \mathbf{\hat{x}} = 
    \left[
        \begin{matrix}
            \hat{x}(1) \\ 
            \hat{x}(2) \\ 
            \vdots \\
            \hat{x}(N)
        \end{matrix}
    \right]
    \in \mathbb{R}^{nN}
\end{equation}

\noindent
and then the sum equation

\begin{equation}
    \sum^{N-1}_{k=0} \Vert \hat{u}(k) \Vert^2_R = \hat{u}^T(0) R \hat{u}(0) + \hat{u}^t(1)R\hat{u}(1) + \dotsm + \hat{u}^T (N-1) R \hat{u}(N-1)
\end{equation}

\noindent
becomes

\begin{equation}
    \underbrace{
        \left[
            \begin{matrix}
                \hat{u}^T(0) & \hat{u}^T(1) & \dotsi & \hat{u}^T(N - 1) 
            \end{matrix}
        \right]
    }_{\mathbf{\hat{u}}^T}
    \left[
        \begin{matrix}
            R & 0 & \dotsi & 0 \\
            0 & R & \dotsi & 0 \\
            \vdots & \vdots & \ddots & \vdots \\
            0 & 0 & \dotsi & R \\
        \end{matrix}
    \right]
    \underbrace{
        \left[
            \begin{matrix}
                \hat{u}(0) \\
                \hat{u}(1) \\
                \vdots \\
                \hat{u}(N - 1)
            \end{matrix}
        \right]
    }_{\mathbf{\hat{u}}}
\end{equation}

Using the notation

\begin{equation}
    \mathbf{R} = \left[
        \begin{matrix}
            R & 0 & \dotsi & 0 \\
            0 & R & \dotsi & 0 \\
            \vdots & \vdots & \ddots & \vdots \\
            0 & 0 & \dotsi & R 
        \end{matrix}
    \right] = diag_N(R)
\end{equation}

\noindent
we get to 

\begin{equation}
    \sum_{k=0}^{N-1} \norma{\hat{u}(k)}{R} = \norma{\mathbf{\hat{u}}}{\mathbf{R}}
\end{equation}

In a similar way, we can write

\begin{equation}
    \begin{gathered}
        \norma{\hat{x}(N)}{P_f} + \sum_{k=0}^{N-1} \norma{\hat{x}(k)}{Q} = \hat{x}^T(0) Q x(0) \\
        + \hat{x}^T(1)Q\hat{x}(1) + \dotsi + \hat{x}^T(N-1)Q\hat{x}(N-1) + \hat{x}^T(N)P_f\hat{x}(N) \\
        = \hat{x}^T(0)Qx(0) +  \\
        \underbrace{
            \left[
                \begin{matrix}
                    \hat{x}^T(1) & \hat{x}^T(2) & \dotsi & \hat{x}^T(N)		
                \end{matrix}
            \right]
        }_{\mathbf{\hat{x}}^T}
        \left[
            \begin{matrix}
                Q & 0 & \dotsi & 0 \\
                0 & Q & \dotsi & 0 \\
                \vdots & \vdots & \ddots & \vdots \\
                0 & 0 & \dotsi & P_f
            \end{matrix}
        \right]
        \underbrace{
            \left[
                \begin{matrix}
                    \hat{x}(1) \\
                    \hat{x}(2) \\
                    \vdots \\
                    \hat{x}(N)
                \end{matrix}
            \right]
        }_{\mathbf{\hat{x}}}
    \end{gathered}
\end{equation}

\noindent
with the notation

\begin{equation}
    Q = 
    \left[
        \begin{matrix}
            Q & 0 & \dotsi & 0 \\
            0 & Q & \dotsi & 0 \\
            \vdots & \vdots & \ddots & \vdots \\
            0 & 0 & \dotsi & P_f
        \end{matrix}
    \right]
    = 
    \left[
        \begin{matrix}
            diag_{N-1}(Q) & 0 \\
            0 & P_f
        \end{matrix}
    \right]
    \in \mathbb{R}^{nN \times nN}
\end{equation}

\noindent
as

\noindent
\begin{equation}
    \norma{\hat{x}(N)}{{_f}} + \sum^{N - 1}_{k=0} \norma{\hat{x}(k)}{Q} = \norma{x(0)}{Q} + \norma{\mathbf{\hat{x}}}{\mathbf{Q}} 
\end{equation}

So, the cost function can be written in the form

\noindent
\begin{equation}
    J(\mathbf{\hat{x}}, \mathbf{\hat{u}}) = \norma{x(0)}{Q} + \norma{\mathbf{\hat{x}}}{\mathbf{Q}} + \norma{\mathbf{\hat{u}}}{\mathbf{R}}
\end{equation}
