Consider a system with discrete-time dynamics described by

    \begin{equation}
        x[k + 1] = A x[k] + B u[k]
        \label{cap3-dynamic-equation}
    \end{equation}

\noindent
where \highlight{$x[k] \in \mathbb{R}^n$} and \highlight{$u[k] \in \mathbb{R}^p$} are 
the state and control vectors at the $k$th time instant.

The goal is to drive the state \highlight{$x[k]$} to the origin from a given initial condition while respecting
state and control constraints in the form

\begin{equation}
    \label{cap3_x_in_X}
    x[k] \in \mathcal{X}
\end{equation}
    
\begin{equation}
    \label{cap3_u_in_U}
    u[k] \in \mathcal{U}
\end{equation}

\noindent
where $\mathcal{U}$ and $\mathcal{X}$ are polyhedral sets containing the origin in their interior.
It is assumed that $\mathcal{U}$ and $\mathcal{X}$ are described in half-space representation as


\begin{equation}
    \mathcal{U} = \left\{
        u \in \mathbb{R}^p : S_u u \leq b_u	
    \right\}
\end{equation}

\noindent
and

\begin{equation}
    \label{cap3_eq_Sx_bx}
    \mathcal{X} = \left\{
        x \in \mathbb{R}^n : S_x x \leq b_x
    \right\}
\end{equation}

\noindent
with \highlight{$S_u$, $b_u$} and \highlight{$S_x$, $b_x$} of suitable dimensions.

\section{Cost Function}


At each time $k$, the control action $u[k]$ will be determined by solving an 
an optimization problem with the cost given by


\begin{equation}
    J = \blue{\norma{\hat{x}[k + N | k]}{P_f}} + \orange{\sum^{N-1}_{i=0} \Big[ \norma{\hat{x}[k + i | k]}{Q} + \norma{\hat{u}[k + i | k]}{R} \Big]}
\end{equation}

\noindent
where $\hat{x}[k+i|k]$ are predicted values of the state and control vectors, which are 
related through the following equations:

\noindent
\begin{equation}
    \hat{x}[k|k] = x[k]
\end{equation}

\noindent
\begin{equation}
    \hat{x}[k + i + 1|k] = A \hat{x}[k+i|k] + B
\end{equation}

\noindent
\begin{equation}
    \hat{x}[k + i|k], \quad i =  0, 1, \dotsm, N-1
\end{equation}


The expression for \highlight{$J$} consists of a \blue{terminal cost} term and a trajectory \orange{cost term}. 

With \highlight{$P_f$}, \highlight{$Q$}, \highlight{$R$} being symmetric and positive-definite, 
with \highlight{$P_f$} chosen so that the terminal cost term corresponds 
to the value function of an infinite horizon Discrete-Time Linear Quadratic Regulator (DLQR) designed width
weights $Q$, $R$.
