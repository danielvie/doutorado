    \textbf{Pergunta:} Para m dado N, como determinar o conjunto $\mathcal{D} \subset \mathcal{X}$ de condiþ§es iniciais
    $x(0)$ a partir das quais o conjunto alvo $\mathcal{X}_f$ pode ser atingido repeitando as restricoes sobre o estado e o controle ?
    
    \textbf{Alternativamente:} Como determinal o conjunto $\mathcal{D}$ de condicoes iniciais $x(0)$ para as quais o 
    problema de otimizacao e factivel
    
    $\mathcal{D}$ Ú o conjunto (ou regiao) de pontos $x(0) \in \mathcal{X}$ para os quais exist $\mathbf{u} \in \mathbb{R}^{pN}$ 
    que satisfaz as sequintes restricoes:
    
    \begin{equation}
        \left[ \begin{matrix} \mathbf{S_x} \mathbf{B} \\ \mathbf{S_u} \end{matrix} \right] \mathbf{u} \le
        \left[ \begin{matrix} \mathbf{b_x - S_xA}x(0) \\ \mathbf{b_u} \end{matrix} \right]
    \end{equation}
    
    Essas restricoes podem ser reescritas como

    \begin{equation}
        \left[ \begin{matrix} \mathbf{S_x} \mathbf{B} & \mathbf{S_xA} \\ \mathbf{S_u} & 0 \end{matrix} \right] 
        \left[ \begin{matrix} \mathbf{u} \\ x(0) \end{matrix} \right] \le
        \left[ \begin{matrix} \mathbf{b_x} \\ \mathbf{b_u} \end{matrix} \right]
    \end{equation}

    ou ainda, acrescentando-se a restricao $S_x x(0) \le b_x$,
    ou ainda, acrescentando-se a restricao $S_x x(0) \le b_x$,

    \begin{equation}
        \left[ \begin{matrix} \mathbf{S_x} \mathbf{B} & \mathbf{S_xA} \\ \mathbf{S_u} & 0 \\ 0 & S_x \end{matrix} \right] 
        \left[ \begin{matrix} \mathbf{u} \\ x(0) \end{matrix} \right] \le
        \left[ \begin{matrix} \mathbf{b_x} \\ \mathbf{b_u} \\ b_x \end{matrix} \right]
    \end{equation}
    
    De forma resumida, pode-se escrever
    
    \begin{equation}
        S_w w \le b_w
    \end{equation}
    
    com
    
    \begin{equation}
        w = \left[ \begin{matrix} \mathbf{u} \\ x(0) \end{matrix} \right],
        S_w = \left[ \begin{matrix} \mathbf{S_x} \mathbf{B} & \mathbf{S_xA} \\ \mathbf{S_u} & 0 \\ 0 & S_x \end{matrix} \right],
        b_w = \left[ \begin{matrix} \mathbf{b_x} \\ \mathbf{b_u} \\ b_x \end{matrix} \right]
    \end{equation}

    A restricao $S_w w \le b_w$ define um poliedro $\mathcal{P}_w$ no espaco $\mathbb{R}^{(pN+n)}$ correspondente as variaveis 
    \textbf{u}, x(0).
    
    O conjunto $\mathcal{D}$ e obtido \textbf{projetando-se} $\mathcal{P}_w$ sobre o subespaco de suas ultimas $n$ componentes.
