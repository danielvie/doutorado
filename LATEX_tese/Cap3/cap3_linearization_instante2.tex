Repeating the procedure for $x(t_2)$ we have

\begin{align}
	\label{eq_cap2_lin_x2}
	x(t_2)  &= e^{A_1(t_2-t_1)}x(t_1) + \intlinAA{1}{2}{}{} \\
	\label{eq_cap2_lin_xr2}
	x(\bar{t}_2) &= e^{A_1(\bar{t}_2-\bar{t}_1)}x(\bar{t}_1) + \intlinAbtt{1}{2}
\end{align}

subtracting \ref{eq_cap2_lin_x2} and \ref{eq_cap2_lin_xr2} 

\begin{equation}
	\label{eq_cap2_lin_erro_x1x2}
	\begin{split}
		x(t_2) - x(\bar{t}_2) = 
		\overbrace{\green{e^{A_1(t_2-t_1)}x(t_1) - e^{A_1(\bar{t}_2-\bar{t}_1)}x(\bar{t}_1)}}^{\green{\square}} \\ 
		+ \underbrace{\orange{\intlinAA{1}{2}{}{} - \intlinAbtt{1}{2}}}_{\orange{\bigcirc}}
	\end{split}
\end{equation}

expanding $\green{\square}$

% 2.10
\begin{align}
	\green{ \square } &=
	\green{
		e^{A_1(\bar{t}_2 + \delta t_2 - \bar{t}_1 - \delta t_1)} x(t_1)
		- e^{A_1(\bar{t}_2 - \bar{t}_1)} \bar{x}(\bar{t}_1)
	} \nonumber \\
	&= \green{
		\blue{\Fri{1}{2}} \purple{e^{A_1(\delta t_2 - \delta t_1)}} x(t_1)
		- \blue{\Fri{1}{2}} \bar{x}(\bar{t}_1)
	} \nonumber \\
	&= \green{
		\blue{\bar{F}_1} \purple{[I + A_1(\delta t_2 - \delta t_1)]} x(t_1)
		- \blue{\bar{F}_1} \bar{x}(\bar{t}_1)
	} \nonumber \\
	\label{eq_cap2_lin_x2_green}
	&= \green{
		\bar{F}_1 [x(t_1) - \bar{x}(\bar{t}_1)] + \bar{F}_1 A_1 x(t_1) \delta t_2
		- \bar{F}_1 A_1 x(t_1) \delta t_1
	}
\end{align}

expanding $\orange{\bigcirc}$

% 2.11
\begin{align}
	\orange{\bigcirc} 	&= \orange{\intlinAA{1}{2}{}{} - \intlinAbtt{1}{2}} \nonumber \\
					   	&= \orange{ \intlinCb{\bar{t}_1+\delta t_1}{\bar{t}_2+\delta t_2}{1}{2} - \intlinAbtt{1}{2}} \nonumber \\
						\begin{split}
							&= \orange{ \purple{ e^{A_1 \delta t_2} }
								\bigg[
									\blue{\intlinAbtt{1}{2}} - \intlinBb{\bar{t}_1}{\bar{t}_1 + \delta t_1}{1}{2} + \intlinBb{\bar{t}_2}{\bar{t}_2 + \delta t_2}{1}{2} \bigg] } \\
							& \qquad \orange{ - \blue{\intlinAbtt{1}{2}} }
						\end{split} \nonumber \\
						&= \orange{ \purple{(I + A_1 \delta t_2)}
							\bigg[
								\blue{\bar{G}_1} - \pink{\intlinBb{\bar{t}_1}{\bar{t}_1 + \delta t_1}{1}{2}} + \brown{\intlinBb{\bar{t}_2}{\bar{t}_2 + \delta t_2}{1}{2}}
							\bigg] - \blue{\bar{G}_1}
						} \nonumber \\
						&= \orange{ (I + A_1 \delta t_2)
							\bigg[
								\bar{G}_1 - \pink{e^{A_1(\bar{t}_2 - \bar{t}_1)} B_1 \delta t_1} + \brown{B_1 \delta t_2}
							\bigg] - \bar{G}_1
						} \nonumber \\
						&= \orange{ (I + A_1 \delta t_2)
							\bigg[
								\bar{G}_1 - \pink{\bar{F}_1 B_1 \delta t_1} + B_1 \delta t_2
							\bigg] - \bar{G}_1
						} \nonumber \\
						&= \orange{ \cancel{\bar{G}_1} - \bar{F}_1 B_1 \delta t_1 + B_1 \delta t_2 + A_1 \bar{G}_1 \delta t_2 - \cancel{\bar{G}_1} } \nonumber \\
						\label{eq_cap2_lin_x2_orange}
						&= \orange{ A_1 \bar{G}_1 \delta t_2 + B_1 \delta t_2 - \bar{F}_1 B_1 \delta t_1 }
\end{align}

\noindent replacing (\ref{eq_cap2_lin_x2_green}) and (\ref{eq_cap2_lin_x2_orange}) into equation (\ref{eq_cap2_lin_erro_x1x2})
\begin{equation}
	\label{eq_xt2_xt1_0}
	\begin{split}
		x(t_2) - \bar{x}(\bar{t}_2) =  & \bar{F}_1 [\green{x(t_1) - x_{r_1}(\bar{t}_1)}] \\
							     & \quad - \bar{F}_1 [A_1 \blue{x(t_1)} + B_1] \delta t_1 \\
							     & \quad + [ \bar{F}_1 A_1 \underbrace{\blue{x(t_1)}}_{\blue{\triangle}} + A_1 \bar{G}_1 + B_1] \delta t_2
	\end{split} 
\end{equation}

\noindent expanding $\blue{\triangle}$ from \ref{eq_cap2_lin_erro_x0x1_resultado}

\begin{equation}
	\blue{\triangle} = x(t_1) = \bar{x}(\bar{t}_1) + \bar{F}_0 [ x(t_0) - \bar{x}(t_0) ] + [ \bar{F}_0 A_0 x(t_0) + A_0 \bar{G}_0 + B_0 ] \delta t_1 
	\nonumber
\end{equation}

note that when $\blue{\triangle}$ is replaced in eq. \ref{eq_xt2_xt1_0} it will produce a product with $\delta t_1$ and $\delta t_2$ like
\begin{equation}
	x(t_1)\delta t_1 = \bar{x}(\bar{t}_1)\delta t_1 + \bar{F}_0 \red{\overbrace{ [ x(t_0) - \bar{x}(t_0) ]\delta t_1 }^{\square}} + [ \bar{F}_0 A_0 x(t_0) + A_0 \bar{G}_0 + B_0 ] \purple{\overbrace{\delta t_1\delta t_1}^{\bigcirc}} 
	\nonumber
\end{equation}

both $\red{\square}$ and $\purple{\bigcirc}$ are products of differences that produces a smaller number and they will be neglected in the first order aproximations

\begin{equation}
	\label{eq_xt1_deltat1_aprox}
	x(t_1)\delta t_1 \doteq \bar{x}(\bar{t}_1)\delta t_1 
\end{equation}

replacing equation \ref{eq_cap2_lin_erro_x0x1_resultado} and \ref{eq_xt1_deltat1_aprox} in \ref{eq_xt2_xt1_0}
\begin{equation}
	\begin{split}
		x(t_2) - \bar{x}(\bar{t}_2) =  & \bar{F}_1 \{\green{
										\bar{F}_0 [x(t_0) - \bar{x}(t_0) + [\bar{F}_0 A_0 x(t_0) + A_0 \bar{G}_0 + B_0] \delta t_1]
								 }\} \\
							     & \quad - \bar{F}_1 [A_1 \blue{\bar{x}(t_1)} + B_1] \delta t_1 \\
							     & \quad + [ \bar{F}_1 A_1 \blue{\bar{x}(t_1)} + A_1 \bar{G}_1 + B_1] \delta t_2
	\end{split}
	\nonumber
\end{equation}
\begin{equation}
	\begin{split}
		x(t_2) - \bar{x}(\bar{t}_2) =  & \bar{F}_1 \bar{F}_0 [x(t_0) - \bar{x}(t_0)] \\
							     & \quad + \bar{F}_1 [\bar{F}_0 A_0 x(t_0) + A_0 \bar{G}_0 - A_1 \bar{x}(\bar{t}_1) + B_0 - B_1] \delta t_1 \\
							     & \quad + [\bar{F}_1 A_1 \bar{x}(\bar{t}_1) + A_1 \bar{G}_1 + B_1] \delta t_2
	\end{split}
\end{equation}
