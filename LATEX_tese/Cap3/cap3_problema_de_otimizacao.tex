
\section{Problema de Otimizaçăo}

O problema de otimizaçăo a ser resolvido pode ser expresso como

\begin{equation}
	\min_{\mathbf{x} \in \mathbb{R}^{nN}, \mathbf{u} \in \mathbb{R}^{pN}} J(x,u)
\end{equation}

sendo

\begin{equation}
	J(\mathbf{x}, \mathbf{u}) = \norma{x(0) - \bar{x}}{Q} + \norma{\mathbf{x} - col_N(\bar{x})}{\mathbf{Q}} + \norma{\mathbf{u} - col_N(\bar{u})}{\mathbf{R}}
\end{equation}

com

\begin{equation}
    \mathbf{u}(k) = \left[
        \begin{matrix}
            u(k|k) \\
            u(k + 1|k) \\
            \vdots \\ 
            u(k + N - 1|k)
        \end{matrix}
    \right] \in \mathbb{R}^{pN},
    \mathbf{x}(k) = \left[
        \begin{matrix}
            x(k + 1|k) \\
            x(k + 2|k) \\
            \vdots \\ 
            x(k + N|k)
        \end{matrix}
    \right] \in \mathbb{R}^{nN},
\end{equation}

sujeitas a

\begin{equation}
	\mathbf{S_x} \mathbf{x}(k) \leq \mathbf{b_x}
\end{equation}

\begin{equation}
	\mathbf{S_u} \mathbf{u}(k) \leq \mathbf{b_u}
\end{equation}

Os vetores $\mathbf{x}(k)$ e $\mathbf{u}(k)$ estăo vinculados pela seguinte \textbf{equaçăo de preciçăo}:

\begin{equation}
    \mathbf{x}(k) = \mathbf{A} x(k) + \mathbf{B} \mathbf{u}(k)
\end{equation}

Com isso, o custo pode ser reescrito como

\begin{gather}
        J(\mathbf{u}(k)) = \mathbf{u}(k)^T(\mathbf{B}^T \mathbf{Q} \mathbf{B} + \mathbf{R})\mathbf{u}(k) \nonumber \\
        2 \Bigg\{
            [ \mathbf{A} x(k) - col_N(\bar{x}) ]^T \mathbf{Q} \mathbf{B} - [col_N(\bar{u})]^T \mathbf{R}
        \Bigg\} \mathbf{u}(k) + c_0
         \nonumber
\end{gather}

com

\begin{equation}
    c_0 = \norma{x(k) - \bar{x}}{Q} + \norma{\mathbf{A} x(k) - col_N(\bar{x})}{\mathbf{Q}} + \norma{col_N(\bar{u})}{\mathbf{R}}
    \nonumber
\end{equation}

e as restriçőes passam a ser expressas na forma

\begin{equation}
    \mathbf{S} \mathbf{u}(k) \leq \mathbf{b}
    \nonumber
\end{equation}

com 

\begin{equation}
    \mathbf{S} = \left[
        \begin{matrix}
            \mathbf{S_x} \mathbf{B} \\
            \mathbf{S_u}
        \end{matrix}
    \right], \mathbf{b} = \left[
        \begin{matrix}
            \mathbf{b_x} - \mathbf{S_x} \mathbf{A} x(k) \\
            \mathbf{b_u}
        \end{matrix}
    \right]
    \nonumber
\end{equation}

Tendo-se obtido a soluçao $\mathbf{u}(k)$ para o problema $\mathbb{P}\Big(x(k)\Big)$, o controle a ser aplicado ŕ planta é dado por

\begin{equation}
    u(k) = \hat{u}^*(k|k)
\end{equation}
