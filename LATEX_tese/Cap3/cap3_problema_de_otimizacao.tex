
\section{Optimization Problem}

The optimization problem can be expressed as

\begin{equation}
	\min_{\mathbf{x} \in \mathbb{R}^{nN}, \mathbf{u} \in \mathbb{R}^{pN}} J(x,u)
\end{equation}

where

\begin{equation}
	J(\mathbf{x}, \mathbf{u}) = \norma{x(0) - \bar{x}}{Q} + \norma{\mathbf{x} - col_N(\bar{x})}{\mathbf{Q}} + \norma{\mathbf{u} - col_N(\bar{u})}{\mathbf{R}}
\end{equation}

with

\begin{equation}
    \mathbf{\hat{u}}(k) = \left[
        \begin{matrix}
            \hat{u}(k|k) \\
            \hat{u}(k + 1|k) \\
            \vdots \\ 
            \hat{u}(k + N - 1|k)
        \end{matrix}
    \right] \in \mathbb{R}^{pN},
    \mathbf{\hat{x}}(k) = \left[
        \begin{matrix}
            \hat{x}(k + 1|k) \\
            \hat{x}(k + 2|k) \\
            \vdots \\ 
            \hat{x}(k + N|k)
        \end{matrix}
    \right] \in \mathbb{R}^{nN},
\end{equation}

subject to

\begin{equation}
	\mathbf{S_x} \mathbf{x}(k) \leq \mathbf{b_x}
\end{equation}

\begin{equation}
	\mathbf{S_u} \mathbf{u}(k) \leq \mathbf{b_u}
\end{equation}

    \mathbf{x}(k) = \mathbf{A} x(k) + \mathbf{B} \mathbf{u}(k)
\end{equation}

Thus, the cost can be rewritten as

\begin{gather}
        J(\mathbf{u}(k)) = \mathbf{u}(k)^T(\mathbf{B}^T \mathbf{Q} \mathbf{B} + \mathbf{R})\mathbf{u}(k) \nonumber \\
        2 \Bigg\{
            [ \mathbf{A} x(k) - col_N(\bar{x}) ]^T \mathbf{Q} \mathbf{B} - [col_N(\bar{u})]^T \mathbf{R}
        \Bigg\} \mathbf{u}(k) + c_0
         \nonumber
\end{gather}

with

\begin{equation}
    c_0 = \norma{x(k) - \bar{x}}{Q} + \norma{\mathbf{A} x(k) - col_N(\bar{x})}{\mathbf{Q}} + \norma{col_N(\bar{u})}{\mathbf{R}}
    \nonumber
\end{equation}

}
and the constraints can be expressed as

\begin{equation}
    \mathbf{S} \mathbf{u}(k) \leq \mathbf{b}
    \nonumber
\end{equation}

with

\begin{equation}
    \mathbf{S} = \left[
        \begin{matrix}
            \mathbf{S_x} \mathbf{B} \\
            \mathbf{S_u}
        \end{matrix}
    \right], \mathbf{b} = \left[
        \begin{matrix}
            \mathbf{b_x} - \mathbf{S_x} \mathbf{A} x(k) \\
            \mathbf{b_u}
        \end{matrix}
    \right]
    \nonumber
\end{equation}

Having obtained the solution $\mathbf{u}(k)$ for the problem $\mathbb{P}\Big(x(k)\Big)$, the control to be applied to the plant is given by

\begin{equation}
    u(k) = \hat{u}^*(k|k)
\end{equation}

where $\hat{u}^*(k|k)$ corresponds to the first $p$ elements of $\mathbf{\hat{u}}^*(k)$.
