\newcommand{\intlinAbft}[2]
{
    \int_{t_#1}^{\bar{t}_#2}
    {e^{A_#2(\bar{t}_#2 - \tau)} B_#2 d \tau}
}
% #1: idx parte de baixo da integral
% #2: idx parte de cima da integral
%
% ex: \intlinAb11{1}{2}
% integral barra false true

\newcommand{\intlinAA}[4]
{
    \int_{t_{#3_#1}}^{t_{#4_#2}}
    {e^{A_#2(t_{#2} - \tau)} B_#2 d \tau}
}
% mesmo que `\intlinAA`, porem com: exp(Ai(ti - tau)) ao inves de exp(Ai(tri - tau))
%
% ex: \intlinAA{1}{2}{r}{r}

\newcommand{\intlinB}[4]
{
    \int_{#1}^{#2}
    {e^{A_#4(\bar{t}_{#4} - \tau)} B_#4 d \tau}
}
% #1: parte de baixo da integral
% #2: parte de cima da integral
% #3, #4: idx -> exp(A#3(Tr#4 - tau)) B#3
%
% ex: \intlinB{t_{r_1}}{t_{r_1}+\delta t_1}{1}{2}

\newcommand{\intlinBb}[4]
{
    \int_{#1}^{#2}
    {e^{A_#4(\bar{t}_#4 - \tau)} B_#4 d \tau}
}
% #1: parte de baixo da integral
% #2: parte de cima da integral
% #3, #4: idx -> exp(A#3(Tr#4 - tau)) B#3
%
% ex: \intlinB{t_{r_1}}{t_{r_1}+\delta t_1}{1}{2}

\newcommand{\intlinCb}[4]
{
    \int_{#1}^{#2}
    {e^{A_#4(\bar{t}_#4 + \delta t_#4 - \tau)} B_#4 d \tau}
}
% #1: parte de baixo da integral
% #2: parte de cima da integral
% #3, #4: idx -> exp(A#3(Tr#4 + \delta#4 - tau)) B#3
%
% ex: \intlinC{t_{r_1}}{t_{r_1}+\delta t_1}{1}{2}

\newcommand{\intlinG}[4]
{
    \int_{#1}^{#2}
    {e^{A#4(#3 - \tau)} B#4 d\tau}
}
% #1: parte de baixo da integral
% #2: parte de cima da integral
% #3: dentro da exponencial
% _#4: nro de Ai e Bi
% exemplo: \intlinG{0}{t}{t}{_0}
