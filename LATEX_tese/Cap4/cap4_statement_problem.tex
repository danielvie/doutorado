Considering a plant of an affine switched system described by: 

\begin{equation}
    \label{eq_continuous_time_dynamics}
    \dot{x}(t) = A_{\sigma(t)} x(t) + b_{\sigma(t)}
\end{equation}

The control action applied in the system is in the form of perturbations in each operation mode switching time. By allowing the control the 
a system using predictive control, we need to find a linear equation that relates the switching times and the resulting system at the end of 
one cycle.  

\subsection{Cost Function}

\begin{equation}
\label{eq_cost_funcion}
    J = \norma{e[N_p]}{P_f} \\
    + \sum^{N_p - 1}_{j=0}\norma{e[j]}{Q} + \norma{\delta t[j]}{R} 
\end{equation}


\begin{equation}
    e(t_{j,N}) = \Phi e(t_{j,0}) + \Gamma \delta t[j]
\end{equation}

em que $Q = Q^T > 0$, $\rho > 0$

\begin{equation}
    \delta t = [t_{j,1} - \bar{t}_{j,1}, t_{j,2} - \bar{t}_{j,2}, ...]^T
\end{equation}

\subsection{Dwell-time Constraints}

We want to enforce a constraint on the dwell times where the switching times will the subject to the 
constraints:

\noindent
\begin{equation}
    (t_i - t_{i - 1}) \ge \Delta t_{i,min}\;, \quad i = 1,2,\dotso,N
\end{equation}

\noindent
then we can write the restriction as

\noindent
\begin{align}
        \delta t_1 &\ge - \Delta \bar{t}_1 + \Delta t_{1,min} \\
        \delta t_i - \delta t_{i-1} &\ge - \Delta \bar{t}_i + \Delta t_{i,min}\;, \quad i = 1,2,\dotso,N \\
        \delta t_{N-1} &\ge - \Delta \bar{t}_N + \Delta t_{N,min}
\end{align}

\noindent
\begin{equation}
    \Delta \bar{t}_i = \bar{t}_i - \bar{t}_{i-1}
\end{equation}

\noindent
to match the constraint from \eqref{cap3-constraint} 

\begin{equation*}
    \mathbf{S} \mathbf{\hat{u}} [k] \le b
    \nonumber
\end{equation*}


\noindent
the equations can be written as

\noindent
\begin{align}
        -\delta t_1 &\le + \Delta \bar{t}_1 - \Delta t_{1,min} \\
        -\delta t_i + \delta t_{i-1} &\le + \Delta \bar{t}_i - \Delta t_{i,min}\;, \quad i = 1,2,\dotso,N \\
        -\delta t_{N-1} &\le + \Delta \bar{t}_N - \Delta t_{N,min}
\end{align}

\noindent
with 

\begin{equation}
    \mathbf{L} = 
    \left[
        \begin{matrix}
            -1 & 0 & 0 & \dotsi &  0 &  0 \\
            1 & -1 & 0 & \dotsi &  0 &  0 \\
            \vdots & \vdots & \vdots & \ddots & \vdots & \vdots \\
            0 & 0 & 0 & \dotsi & 1 &  -1 \\
            0 & 0 & 0 & \dotsi &  0 & 1 \\
        \end{matrix}
    \right]
\end{equation}

\begin{equation}
    c = 
    \left[
        \begin{matrix}
            \Delta t_{r_1} -\Delta t_{r_{1, min}} \\
            \Delta t_{r_2} -\Delta t_{r_{2, min}} \\
            \vdots \\
            \Delta t_{r_{N-1}} -\Delta t_{r_{{N-1}, min}} \\
            \Delta t_{r_N} -\Delta t_{r_{N, min}} \\
        \end{matrix}
    \right]
\end{equation}
