Repeating the procedure for $x(t_3)$ we have

\noindent
\begin{align}
	\label{eq_cap3_lin_x3}
	x(t_3)  &= e^{A_3(t_3-t_2)}x(t_2) + \intlinAA{2}{3}{}{} \\
	\label{eq_cap3_lin_xr3}
	\bar{x}(\bar{t}_3) &= e^{A_3(\bar{t}_3-\bar{t}_2)}\bar{x}(\bar{t}_2) + \intlinAbtt{2}{3}
\end{align}

\noindent
subtracting \eqref{eq_cap3_lin_x3} from \eqref{eq_cap3_lin_xr3} 

\noindent
\begin{equation}
	\label{eq_cap3_lin_erro_x2x3}
	\begin{split}
		x(t_3) ->  - \bar{x}(\bar{t}_3) = \green{\overbrace{e^{A_3(t_3-t_2)}x(t_2) - e^{A_3(\bar{t}_3-\bar{t}_2)}\bar{x}(\bar{t}_2)}^{\square}} \\ 
		+ \orange{\underbrace{\intlinAA{2}{3}{}{} - \intlinAbtt{2}{3}}_{\bigcirc}}
	\end{split}
\end{equation}

\noindent 
expanding $\green{\square}$ in \eqref{eq_cap3_lin_erro_x2x3}:

\noindent
\begin{align}
	\green{ \square } &=
	\green{
		e^{A_3(\bar{t}_3 + \delta t_3 - \bar{t}_2 - \delta t_2)} x(t_2)
		- e^{A_3(\bar{t}_3 - \bar{t}_2)} \bar{x}(\bar{t}_2)
	} \nonumber \\
	&= \green{
		\blue{\Fri{2}{3}} \purple{e^{A_3(\delta t_3 - \delta t_2)}} x(t_2)
		- \blue{\Fri{2}{3}} \bar{x}(\bar{t}_2)
	} \nonumber \\
	&= \green{
		\blue{\bar{F}_3} \purple{[I + A_3(\delta t_3 - \delta t_2)]} x(t_2)
		- \blue{\bar{F}_3} \bar{x}(\bar{t}_2)
	} \nonumber \\
	\label{eq_cap3_lin_x3_green}
	&= \green{
		\bar{F}_3 [x(t_2) - \bar{x}(\bar{t}_2)] + \bar{F}_3 A_3 x(t_2) \delta t_3
		- \bar{F}_3 A_3 x(t_2) \delta t_2
	} 
\end{align}

\noindent 
and expanding $\orange{\bigcirc}$:

% 2.17
\noindent
\begin{align}
	\orange{\bigcirc} 	&= \orange{ \intlinAA{2}{3}{}{} - \intlinAbtt{2}{3} } \nonumber \\
					   	&= \orange{ \intlinCb{\bar{t}_2+\delta t_2}{\bar{t}_3+\delta t_3}{2}{3} - \intlinAbtt{2}{3}} \nonumber \\
						\begin{split}
							&= \orange{ \purple{ e^{A_3 \delta t_3} }
								\bigg[
									\blue{\intlinAbtt{2}{3}} - \intlinBb{\bar{t}_2}{\bar{t}_2 + \delta t_2}{2}{3} + \intlinBb{\bar{t}_3}{\bar{t}_3 + \delta t_3}{2}{3} \bigg] } \\
							& \qquad \orange{ - \blue{\intlinAbtt{2}{3}} }
						\end{split} \nonumber \\
						&= \orange{ \purple{(I + A_3 \delta t_3)}
							\bigg[
								\blue{ \bar{G}_3 } - \pink{\intlinBb{\bar{t}_2}{\bar{t}_2 + \delta t_2}{2}{3}} + \brown{\intlinBb{\bar{t}_3}{\bar{t}_3 + \delta t_3}{2}{3}}
							\bigg] - \blue{\bar{G}_3}
						} \nonumber \\
						&= \orange{ (I + A_3 \delta t_3)
							\bigg[
								\bar{G}_3 - \pink{e^{A_3(\bar{t}_3 - \bar{t}_2)} B_3 \delta t_2} + \brown{B_3 \delta t_3}
							\bigg] - \bar{G}_3
						} \nonumber \\
						&= \orange{ (I + A_3 \delta t_3)
							\bigg[
								\bar{G}_3 - \pink{\bar{F}_3 B_3 \delta t_2} + B_3 \delta t_3
							\bigg] - \bar{G}_3
						} \nonumber \\
						&= \orange{ \cancel{\bar{G}_3} - \bar{F}_3 B_3 \delta t_2 + B_3 \delta t_3 + A_3 \bar{G}_3 \delta t_3 - \cancel{\bar{G}_3} } \nonumber \\
						\label{eq_cap3_lin_x3_orange}
						&= \orange{ A_3 \bar{G}_3 \delta t_3 + B_3 \delta t_3 - \bar{F}_3 B_3 \delta t_2 }
\end{align}

\noindent 
and then replacing \eqref{eq_cap3_lin_x3_green} in \eqref{eq_cap3_lin_x3_orange} back in eq. \eqref{eq_cap3_lin_erro_x2x3} we get to the difference equation
for $x(t_3) - \bar{x}(\bar{t}_3)$

\noindent
\begin{align}
	\label{eq_xt3_xt2_1_0}
	\begin{split}
		x(t_3) - \bar{x}(\bar{t}_3) =  & \bar{F}_3 [\green{x(t_2) - \bar{x}_2(\bar{t}_2)}] \\
							     & - \bar{F}_3 [A_3 \blue{x(t_2)} + B_3] \delta t_2 \\
							     & + [ \bar{F}_3 A_3 \blue{x(t_2)} + A_3 \bar{G}_3 + B_3] \delta t_3
	\end{split}
\end{align}

\noindent 
from \eqref{eq_cap3_lin_erro_x2x1_resultado}, $\blue{\triangle}$ can be expanded the same way as in the past iteration and $x(t_2)-\bar{x}(\bar{t}_2)$ can be replaced in \eqref{eq_xt3_xt2_1_0} 

\noindent
\begin{align}
	\begin{split}
		x(t_3) - \bar{x}(\bar{t}_3) =  & \bar{F}_3 \{\green{\bar{F}_2\bar{F}_1[x(t_0) - \bar{x}(t_0)]} \\
								 & \green{ + \bar{F}_2[\bar{F}_1 A_1 x(t_0) + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2] \delta t_1 } \\
								 & \green{ + [\bar{F}_2 A_2 \bar{x}(t_1) + A_2 \bar{G}_2 + B_2 ] \delta t_2 } \} \\
							     & - \bar{F}_3 [A_3 \blue{\bar{x}(t_2)} + B_3] \delta t_2 \\
							     & + [ \bar{F}_3 A_3 \blue{\bar{x}(t_2)} + A_3 \bar{G}_3 + B_3] \delta t_3
	\end{split}
	\nonumber
\end{align}

\noindent
after rearranging the terms, one arrives at

\noindent
\boxeq{eq_cap3_lin_erro_x3x2_resultado}
{
	\begin{split}
		x(t_3) - \bar{x}(\bar{t}_3) =  & \bar{F}_3\bar{F}_2\bar{F}_1 [x(t_0) - \bar{x}(t_0)] \\
								 & + \bar{F}_3\bar{F}_2[\bar{F}_1 A_1 x(t_0) + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2] \delta t_1 \\
								 & + \bar{F}_3[\bar{F}_2 A_2 \bar{x}(t_1) + A_2 \bar{G}_2 - A_3 \bar{x}(\bar{t}_2) + B_2 - B_3 ] \delta t_2 \\
							     & + [\bar{F}_3 A_3 \bar{x}(t_2) + A_3 \bar{G}_3 + B_3] \delta t_3
	\end{split}
}
