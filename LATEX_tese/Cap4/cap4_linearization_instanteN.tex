        & + \bar{F}_{i}\bar{F}_{i-1} \ldots \bar{F_1} [\bar{F}_1 A_1 x(t_0) + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2] \delta t_1 \\
        & + \bar{F}_{i}\bar{F}_{i-1} \ldots \bar{F_2} [\bar{F}_2 A_2 x(t_1) + A_2 \bar{G}_2 - A_3 \bar{x}(\bar{t}_2) + B_2 - B_3] \delta t_2 \\
        & \qquad \vdots \\
        & + \bar{F}_{i}[\bar{F}_{i-1} A_{i-1} \bar{x}(t_{i-2}) + A_{i-1} \bar{G}_{i-1} - A_{i} \bar{x}(\bar{t}_{i-1}) + B_{i-1} - B_{i} ] \delta t_{i-1} \\
        & + [\bar{F}_{i} A_{i} \bar{x}(t_{i}) + A_{i} \bar{G}_{i} + B_{i}] \delta t_i
    \end{split}
}

                                 & + \bar{F}_{N}\bar{F}_{N-1} \ldots \bar{F_2} [\green{\overbrace{\bar{F_1} A_1 x(t_0) + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2}^{\square}}] \delta t_1 \nonumber \\
                                 & + \bar{F}_{N}\bar{F}_{N-1} \ldots \bar{F_3} [\green{\bar{F_2} A_2 x(t_1) + A_2 \bar{G}_2 - A_3 \bar{x}(\bar{t}_2) + B_2 - B_3}] \delta t_2 \nonumber \\

                                 \end{split} \nonumber \\

\end{align}

The equation \eqref{eq_xN_propagation} can be developed further to remove the dependencies on the intermediary states 
\highlight{$x(t_0), x(t_1), ..., x(t_{N-2})$}. By expanding $\green{\square}$ and $\orange{\bigcirc}$, the following identity holds

% 2.23
\noindent
\begin{align}
    \label{eq_square_xN}
    \green{\square} &= \green{\bar{F_1} A_1 x(t_0) + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2} \nonumber \\
                    &= \green{A_1 \purple{\bar{F_1} x(t_0)} + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2} \nonumber \\
                    &= \green{A_1 \purple{[\bar{x}(\bar{t}_1) - \bar{G}_1]} + A_1 \bar{G}_1 - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2} \nonumber \\
                    &= \green{A_1 \bar{x}(\bar{t}_1) - \cancel{A_1 \bar{G}_1} + \cancel{A_1 \bar{G}_1} - A_2 \bar{x}(\bar{t}_1) + B_1 - B_2} \nonumber \\
                    &= \green{(A_1 - A_2) \bar{x}(\bar{t}_1) + B_1 - B_2}
\end{align}

% 2.24
\noindent
\begin{align}
    \label{eq_circle_xN}
    \orange{\bigcirc} &= \orange{\bar{F}_{N} A_{N} \bar{x}(t_{N-1}) + A_{N} \bar{G}_{N} + B_{N}} \nonumber \\
                     &= \orange{A_{N} \purple{\bar{F}_{N} \bar{x}(t_{N-1})} + A_{N} \bar{G}_{N} + B_{N}} \nonumber \\
                     &= \orange{A_{N} \purple{[\bar{x}(\bar{t}_N) - \bar{G}_{N}]} + A_{N} \bar{G}_{N} + B_{N}} \nonumber \\
                     &= \orange{A_{N} \bar{x}(\bar{t}_N) - \cancel{A_{N} \bar{G}_{N}} + \cancel{A_{N} \bar{G}_{N}} + B_{N}} \nonumber \\
                     &= \orange{A_{N} \bar{x}(\bar{t}_N) + B_{N}}
\end{align}

\noindent 
replacing \eqref{eq_square_xN} and \eqref{eq_circle_xN} in \eqref{eq_xN_propagation}

                                 & + \bar{F}_{N}\bar{F}_{N-1} \ldots \bar{F_2} [\green{(A_1 - A_2) \bar{x}(\bar{t}_1) + B_1 - B_2}] \delta t_1 \\
                                 & + \bar{F}_{N}\bar{F}_{N-1} \ldots \bar{F_3} [\green{(A_2 - A_3) \bar{x}(\bar{t}_2) + B_2 - B_3}] \delta t_2 \\
                                 & \qquad \vdots \\
                                 & + \bar{F}_{N}[\green{(A_{N-1} - A_{N}) \bar{x}(\bar{t}_{N-1}) + B_{N-1} - B_{N}}] \delta t_{N-1} \\
                                 & + [\orange{A_{N} \bar{x}(\bar{t}_N) + B_{N}}] \delta t_N
    \end{split}
}

\noindent
and renaming the terms of \eqref{eq_cap3_lin_erro_xNx0_resultado} to

\noindent
\begin{align}
    e(t) &= x(t) - \bar{x}(\bar{t}) \\ \nonumber \\
    \Phi &= \bar{F}_{N} \bar{F}_{N-1} \ldots \bar{F_1} \\ \nonumber \\
    \label{eq_Phi}
    \begin{split}
        \Gamma &= \Big[ \bar{F}_{N}\bar{F}_{N-1} \ldots \bar{F_2} [(A_1 - A_2) \bar{x}(\bar{t}_1) + B_1 - B_2], \\
        & \qquad \bar{F}_{N}\bar{F}_{N-1} \ldots \bar{F_3} [(A_2 - A_3) \bar{x}(\bar{t}_2) + B_2 - B_3], \\
        & \qquad \qquad \vdots \\
        & \qquad \bar{F}_{N}[(A_{N-1} - A_{N}) \bar{x}(\bar{t}_{N-1}) + B_{N-1} - B_{N}], \\
        & \qquad A_{N} \bar{x}(\bar{t}_{N}) + B_{N}\Big]
    \end{split} \\ \nonumber \\
    \label{eq_Gamma}
    \delta t &= [\delta t_1, \delta t_2, \ldots, \delta t_N]^T
\end{align}
